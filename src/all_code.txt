use std::io::Write;
use std::time::{Duration, Instant};

use ak680max_driver::device::connection;
use ak680max_driver::protocol::rgb_commands::{
    self, Transport, WIRE_MAGIC_OUT, WIRE_REPORT_SIZE,
};

const SCAN_TIMEOUT_MS: i32 = 150;
const SCAN_DELAY: Duration = Duration::from_millis(30);
const SEND_TIMEOUT_MS: i32 = 1000;

//  helpers 

fn hex_dump(data: &[u8]) -> String {
    let mut s = String::with_capacity(data.len() * 3);
    for (i, b) in data.iter().enumerate() {
        if i > 0 && i % 8 == 0 {
            s.push_str("  ");
        } else if i > 0 {
            s.push(' ');
        }
        s.push_str(&format!("{b:02X}"));
    }
    s
}

fn parse_hex(hex: &str) -> Result<Vec<u8>, String> {
    let clean: String = hex.chars().filter(|c| !c.is_whitespace()).collect();
    if clean.len() % 2 != 0 {
        return Err("Hex string must have even length".into());
    }
    (0..clean.len())
        .step_by(2)
        .map(|i| {
            u8::from_str_radix(&clean[i..i + 2], 16)
                .map_err(|e| format!("Bad hex at position {i}: {e}"))
        })
        .collect()
}

fn parse_hex_byte(s: &str) -> Result<u8, String> {
    let clean = s.trim_start_matches("0x").trim_start_matches("0X");
    u8::from_str_radix(clean, 16).map_err(|e| format!("Bad hex byte '{s}': {e}"))
}

fn build_report(cmd: u8, header: &[u8], data_size: u8) -> [u8; WIRE_REPORT_SIZE] {
    let mut report = [0u8; WIRE_REPORT_SIZE];
    report[0] = WIRE_MAGIC_OUT;
    report[1] = cmd;
    report[2] = data_size;
    for (i, &b) in header.iter().enumerate() {
        if 3 + i < 8 {
            report[3 + i] = b;
        }
    }
    report
}

/// Returns true if the response has non-zero data in bytes 8..32.
fn has_data(response: &[u8; WIRE_REPORT_SIZE]) -> bool {
    response[8..].iter().any(|&b| b != 0)
}

/// Check if an error indicates device disconnect.
fn is_disconnect(e: &ak680max_driver::device::DeviceError) -> bool {
    let msg = format!("{e}");
    msg.contains("0x000001B1") || msg.contains("0x0000048F") || msg.contains("not connected")
}

fn open_device() -> Result<(hidapi::HidDevice, Transport, String), Box<dyn std::error::Error>> {
    let (device, state) = connection::connect()?;
    let transport = state.transport.unwrap_or(Transport::OutputReport);
    let name = state.config.name.to_string();
    Ok((device, transport, name))
}

fn cmd_info() -> Result<(), Box<dyn std::error::Error>> {
    let (_device, transport, name) = open_device()?;
    println!("Device:    {name}");
    println!("Transport: {transport}");
    let (_, state) = connection::connect()?;
    if let Some(ref info) = state.device_info {
        println!("Firmware:  {:.2}", info.firmware_version);
        println!("Battery:   {}%", info.battery_level);
        println!("Charge:    {}", info.charge_status);
        println!("RT Prec:   {}", info.rt_precision);
    }
    Ok(())
}

fn cmd_scan(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    let (device, transport, name) = open_device()?;

    // Parse optional range: scan [start] [end]
    let start = args.first().map(|s| parse_hex_byte(s)).transpose()?.unwrap_or(0x00);
    let end = args.get(1).map(|s| parse_hex_byte(s)).transpose()?.unwrap_or(0x64);

    println!("=== AK680 MAX Protocol Probe ===");
    println!("Device:    {name}");
    println!("Transport: {transport}");
    println!("Range:     {start:#04X}..={end:#04X}");
    println!();

    let mut responsive: Vec<(u8, [u8; WIRE_REPORT_SIZE])> = Vec::new();
    let mut data_cmds: Vec<(u8, [u8; WIRE_REPORT_SIZE])> = Vec::new();
    let scan_start = Instant::now();

    for cmd in start..=end {
        rgb_commands::drain(&device, transport);

        let report = build_report(cmd, &[], 24);

        match rgb_commands::probe_command(&device, &report, transport, SCAN_TIMEOUT_MS) {
            Ok(Some(response)) => {
                if has_data(&response) {
                    println!("[{cmd:#04X}] << {} ***DATA***", hex_dump(&response));
                    data_cmds.push((cmd, response));
                } else {
                    print!(".");
                    std::io::stdout().flush().ok();
                }
                responsive.push((cmd, response));
            }
            Ok(None) => {
                print!("_");
                std::io::stdout().flush().ok();
            }
            Err(e) => {
                if is_disconnect(&e) {
                    println!();
                    println!("!!! Device disconnected at cmd {cmd:#04X}  aborting scan");
                    break;
                }
                println!("[{cmd:#04X}] !! {e}");
            }
        }

        std::thread::sleep(SCAN_DELAY);
    }

    let elapsed = scan_start.elapsed();
    println!();
    println!();
    println!("=== Scan Complete ({:.1}s) ===", elapsed.as_secs_f64());
    println!("{} responded, {} with data", responsive.len(), data_cmds.len());

    if !data_cmds.is_empty() {
        println!();
        println!("Commands with non-zero data:");
        for (cmd, response) in &data_cmds {
            println!("  {cmd:#04X}: {}", hex_dump(response));
        }
    }

    // Show silent ranges.
    if !responsive.is_empty() {
        println!();
        print!("Silent (no response): ");
        let resp_set: std::collections::HashSet<u8> =
            responsive.iter().map(|(c, _)| *c).collect();
        let mut silent = Vec::new();
        for cmd in start..=end {
            if !resp_set.contains(&cmd) {
                silent.push(cmd);
            }
        }
        if silent.is_empty() {
            println!("(none)");
        } else {
            // Print ranges.
            let mut i = 0;
            while i < silent.len() {
                let range_start = silent[i];
                while i + 1 < silent.len() && silent[i + 1] == silent[i] + 1 {
                    i += 1;
                }
                let range_end = silent[i];
                if range_start == range_end {
                    print!("{range_start:#04X} ");
                } else {
                    print!("{range_start:#04X}-{range_end:#04X} ");
                }
                i += 1;
            }
            println!();
        }
    }

    Ok(())
}

fn cmd_send(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    if args.is_empty() {
        println!("Usage: ak680-probe send <hex bytes>");
        println!("  Up to 32 hex bytes, zero-padded. Example:");
        println!("  ak680-probe send AA10180000000000");
        return Ok(());
    }

    let (device, transport, name) = open_device()?;
    println!("Device:    {name}");
    println!("Transport: {transport}");
    println!();

    let hex: String = args.join("");
    let bytes = parse_hex(&hex)?;
    let mut report = [0u8; WIRE_REPORT_SIZE];
    let copy_len = bytes.len().min(WIRE_REPORT_SIZE);
    report[..copy_len].copy_from_slice(&bytes[..copy_len]);

    println!("TX >> {}", hex_dump(&report));

    rgb_commands::drain(&device, transport);

    match rgb_commands::probe_command(&device, &report, transport, SEND_TIMEOUT_MS) {
        Ok(Some(response)) => {
            println!("RX << {}", hex_dump(&response));
            println!();
            print_response_breakdown(&response);
        }
        Ok(None) => println!("RX << (timeout)"),
        Err(e) => println!("RX !! ERROR: {e}"),
    }

    Ok(())
}

fn print_response_breakdown(r: &[u8]) {
    if r.len() < 8 {
        println!("Breakdown: too short ({} bytes)", r.len());
        return;
    }
    println!("Breakdown:");
    println!("  [0]     Magic:   {:#04X}", r[0]);
    println!("  [1]     Command: {:#04X}", r[1]);
    println!("  [2]     Length:  {} ({:#04X})", r[2], r[2]);
    println!(
        "  [3..4]  Addr:    {:#06X} ({})",
        r[3] as u16 | ((r[4] as u16) << 8),
        r[3] as u16 | ((r[4] as u16) << 8),
    );
    println!("  [5..7]  Header:  {} {} {}", r[5], r[6], r[7]);
    println!("  [8..]   Data:");
    for (i, chunk) in r[8..].chunks(8).enumerate() {
        let offset = 8 + i * 8;
        print!("    [{offset:2}] ");
        for b in chunk {
            print!("{b:02X} ");
        }
        print!("  ");
        for b in chunk {
            let c = if b.is_ascii_graphic() { *b as char } else { '.' };
            print!("{c}");
        }
        println!();
    }
}

fn cmd_deep(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    if args.is_empty() {
        println!("Usage: ak680-probe deep <CMD> [header_pos]");
        println!("  Sweep byte 0x00..0xFF at a header position for a given command.");
        println!("  header_pos: 3, 4, 5, 6, or 7 (default: 5, which is header_data[0])");
        println!("  Example: ak680-probe deep 13 5");
        return Ok(());
    }

    let cmd = parse_hex_byte(&args[0])?;
    let pos: usize = args.get(1).map(|s| s.parse()).transpose()?.unwrap_or(5);

    if !(3..=7).contains(&pos) {
        return Err(format!("Position must be 3-7, got {pos}").into());
    }

    let (device, transport, name) = open_device()?;
    println!("Device:    {name}");
    println!("Transport: {transport}");
    println!("Probing cmd={cmd:#04X}, sweeping byte at position [{pos}]");
    println!();

    let mut found = Vec::new();

    for val in 0x00..=0xFFu8 {
        let mut report = [0u8; WIRE_REPORT_SIZE];
        report[0] = WIRE_MAGIC_OUT;
        report[1] = cmd;
        report[2] = 24;
        report[pos] = val;

        rgb_commands::drain(&device, transport);

        match rgb_commands::probe_command(&device, &report, transport, SCAN_TIMEOUT_MS) {
            Ok(Some(response)) => {
                if has_data(&response) {
                    println!("[pos[{pos}]={val:#04X}] << {}", hex_dump(&response));
                    found.push((val, response));
                }
            }
            Ok(None) => {}
            Err(e) => {
                if is_disconnect(&e) {
                    println!("!!! Device disconnected at val={val:#04X}");
                    break;
                }
            }
        }

        std::thread::sleep(SCAN_DELAY);
    }

    println!();
    println!("{} sub-values returned data", found.len());

    Ok(())
}

fn cmd_multi_read(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    if args.len() < 2 {
        println!("Usage: ak680-probe read <CMD> <sub> [chunks]");
        println!("  Read multi-chunk data. Sends with increasing offset.");
        println!("  Example: ak680-probe read 10 00 4");
        println!("  Reads 4 chunks of cmd 0x10 with sub-command 0x00");
        return Ok(());
    }

    let cmd = parse_hex_byte(&args[0])?;
    let sub = parse_hex_byte(&args[1])?;
    let chunks: usize = args.get(2).map(|s| s.parse()).transpose()?.unwrap_or(4);

    let (device, transport, name) = open_device()?;
    println!("Device:    {name}");
    println!("Transport: {transport}");
    println!("Reading cmd={cmd:#04X} sub={sub:#04X} chunks={chunks}");
    println!();

    let mut all_data = Vec::new();

    for chunk in 0..chunks {
        let offset = chunk * 56; // 56 bytes per chunk, not 24
        let mut report = [0u8; WIRE_REPORT_SIZE];
        report[0] = WIRE_MAGIC_OUT;
        report[1] = cmd;
        report[2] = 0x38; // 56 bytes per chunk
        report[3] = (offset & 0xFF) as u8;
        report[4] = ((offset >> 8) & 0xFF) as u8;
        report[5] = sub;

        println!("Chunk {chunk}: TX >> {}", hex_dump(&report));

        rgb_commands::drain(&device, transport);

        match rgb_commands::probe_command(&device, &report, transport, SEND_TIMEOUT_MS) {
            Ok(Some(response)) => {
                println!("         RX << {}", hex_dump(&response));
                all_data.extend_from_slice(&response[8..]); // bytes 8..63 = 56 bytes
            }
            Ok(None) => {
                println!("         RX << (timeout)");
                break;
            }
            Err(e) => {
                println!("         RX !! {e}");
                if is_disconnect(&e) {
                    break;
                }
            }
        }

        std::thread::sleep(SCAN_DELAY);
    }

    if !all_data.is_empty() {
        println!();
        println!("Combined data ({} bytes):", all_data.len());
        for (i, chunk) in all_data.chunks(16).enumerate() {
            let offset = i * 16;
            print!("  {offset:04X}: {:<48}  ", hex_dump(chunk));
            for b in chunk {
                let c = if b.is_ascii_graphic() { *b as char } else { '.' };
                print!("{c}");
            }
            println!();
        }
    }

    Ok(())
}

fn cmd_watch(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    let cmd = args.first().map(|s| parse_hex_byte(s)).transpose()?.unwrap_or(0x13);
    let interval: u64 = args.get(1).map(|s| s.parse()).transpose()?.unwrap_or(500);

    let (device, transport, name) = open_device()?;
    println!("Device:    {name}");
    println!("Watching cmd={cmd:#04X} every {interval}ms (Ctrl+C to stop)");
    println!();

    let report = build_report(cmd, &[], 24);
    let mut last_data = [0u8; WIRE_REPORT_SIZE];
    let mut count = 0u64;

    loop {
        rgb_commands::drain(&device, transport);

        match rgb_commands::probe_command(&device, &report, transport, SEND_TIMEOUT_MS) {
            Ok(Some(response)) => {
                if response != last_data {
                    count += 1;
                    println!("[#{count}] {}", hex_dump(&response));
                    last_data = response;
                }
            }
            Ok(None) => {}
            Err(e) => {
                if is_disconnect(&e) {
                    println!("Device disconnected");
                    break;
                }
            }
        }

        std::thread::sleep(Duration::from_millis(interval));
    }

    Ok(())
}

#[cfg(windows)]
fn cmd_cross(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    if args.is_empty() {
        println!("Usage: ak680-probe cross <hex>");
        return Ok(());
    }

    let api = hidapi::HidApi::new()?;
    let vid: u16 = 0x0C45;
    let pid: u16 = 0x80B2;

    let reader = api
        .device_list()
        .find(|d| d.vendor_id() == vid && d.product_id() == pid && d.usage_page() == 0xFF68)
        .ok_or("Interface 2 (0xFF68) not found")?
        .open_device(&api)?;

    let writer = api
        .device_list()
        .find(|d| d.vendor_id() == vid && d.product_id() == pid && d.usage_page() == 0xFF67)
        .ok_or("Interface 3 (0xFF67) not found")?
        .open_device(&api)?;

    println!("Opened iface 2 (0xFF68) for read, iface 3 (0xFF67) for write");
    println!();

    let hex: String = args.join("");
    let bytes = parse_hex(&hex)?;
    let mut report = [0u8; 64];
    let copy_len = bytes.len().min(64);
    report[..copy_len].copy_from_slice(&bytes[..copy_len]);

    {
        let mut drain_buf = [0u8; 256];
        while reader.read_timeout(&mut drain_buf, 10).unwrap_or(0) > 0 {}
    }

    println!("TX (iface 3) >> {}", hex_dump(&report));

    let mut send_buf = [0u8; 65];
    send_buf[0] = 0x00;
    send_buf[1..].copy_from_slice(&report);
    writer.send_feature_report(&send_buf)?;

    println!("  (sent as feature report, 65 bytes)");

    let mut recv_buf = [0u8; 256];
    let n = reader.read_timeout(&mut recv_buf, 2000)?;

    if n == 0 {
        println!("RX (iface 2) << (timeout - no response)");
    } else {
        let start = if recv_buf[0] == 0x55 { 0 } else if n >= 2 && recv_buf[1] == 0x55 { 1 } else { 0 };
        let avail = n - start;
        let take = avail.min(64);
        println!("RX (iface 2) << {}", hex_dump(&recv_buf[start..start + take]));
        if take >= 8 {
            print_response_breakdown(&recv_buf[start..start + take]);
        }
    }

    Ok(())
}

fn cmd_cross_scan(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    let start = args.first().map(|s| parse_hex_byte(s)).transpose()?.unwrap_or(0x00);
    let end = args.get(1).map(|s| parse_hex_byte(s)).transpose()?.unwrap_or(0x30);

    let api = hidapi::HidApi::new()?;
    let vid: u16 = 0x0C45;
    let pid: u16 = 0x80B2;

    let reader = api
        .device_list()
        .find(|d| d.vendor_id() == vid && d.product_id() == pid && d.usage_page() == 0xFF68)
        .ok_or("Interface 2 (0xFF68) not found")?
        .open_device(&api)?;

    let writer = api
        .device_list()
        .find(|d| d.vendor_id() == vid && d.product_id() == pid && d.usage_page() == 0xFF67)
        .ok_or("Interface 3 (0xFF67) not found")?
        .open_device(&api)?;

    println!("Cross-interface scan: send on iface 3, read on iface 2");
    println!("Range: {start:#04X}..={end:#04X}");
    println!();

    let mut responsive = Vec::new();

    for cmd in start..=end {
        {
            let mut drain_buf = [0u8; 256];
            while reader.read_timeout(&mut drain_buf, 10).unwrap_or(0) > 0 {}
        }

        let mut report = [0u8; 64];
        report[0] = 0xAA;
        report[1] = cmd;
        report[2] = 0x38;

        let mut send_buf = [0u8; 65];
        send_buf[0] = 0x00;
        send_buf[1..].copy_from_slice(&report);

        match writer.send_feature_report(&send_buf) {
            Ok(()) => {}
            Err(e) => {
                println!("[{cmd:#04X}] !! send error: {e}");
                continue;
            }
        }

        let mut recv_buf = [0u8; 256];
        let n = reader.read_timeout(&mut recv_buf, 200).unwrap_or(0);

        if n >= 64 {
            let start_off = if recv_buf[0] == 0x55 { 0 } else if recv_buf[1] == 0x55 { 1 } else { 0 };
            if start_off + 64 <= n {
                let mut out = [0u8; 64];
                out.copy_from_slice(&recv_buf[start_off..start_off + 64]);
                let has_d = out[8..].iter().any(|&b| b != 0);
                if has_d {
                    println!("[{cmd:#04X}] << {} ***DATA***", hex_dump(&out));
                } else {
                    print!(".");
                    std::io::stdout().flush().ok();
                }
                responsive.push((cmd, out));
            }
        } else if n > 0 {
            println!("[{cmd:#04X}] << {n} bytes: {}", hex_dump(&recv_buf[..n]));
        } else {
            print!("_");
            std::io::stdout().flush().ok();
        }

        std::thread::sleep(Duration::from_millis(30));
    }

    println!();
    println!();
    println!("{} responded, {} with data",
        responsive.len(),
        responsive.iter().filter(|(_, r)| r[8..].iter().any(|&b| b != 0)).count(),
    );

    for (cmd, r) in &responsive {
        if r[8..].iter().any(|&b| b != 0) {
            println!("  {cmd:#04X}: {}", hex_dump(r.as_slice()));
        }
    }

    Ok(())
}

#[cfg(windows)]
fn cmd_raw(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    use ak680max_driver::protocol::win_hid::{find_device_path, WinHidDevice};

    let path = find_device_path(0x0C45, 0x80B2, 0xFF67)?;
    println!("Opening iface 3 (0xFF67): {path}");
    let dev = WinHidDevice::open(&path)?;

    let caps = dev.get_caps()?;
    println!("HID Capabilities:");
    println!("  Input Report Size:    {} bytes", caps.input_report_byte_length);
    println!("  Output Report Size:   {} bytes", caps.output_report_byte_length);
    println!("  Feature Report Size:  {} bytes", caps.feature_report_byte_length);
    println!();

    if args.is_empty() {
        println!("Usage: ak680-probe raw <hex>");
        println!("  Sends and receives via FEATURE reports (65 bytes) on iface 3.");
        return Ok(());
    }

    let hex: String = args.join("");
    let bytes = parse_hex(&hex)?;

    let feat_size = caps.feature_report_byte_length as usize; // 65

    // Build 64-byte data, send as feature report (65 with report ID).
    let data_size = feat_size - 1; // 64
    let mut data = vec![0u8; data_size];
    let copy_len = bytes.len().min(data_size);
    data[..copy_len].copy_from_slice(&bytes[..copy_len]);

    println!("TX >> {}", hex_dump(&data[..data.len().min(32)]));

    // Send as SET_FEATURE.
    let mut send_buf = vec![0u8; feat_size];
    send_buf[0] = 0x00; // report ID
    send_buf[1..].copy_from_slice(&data);

    let ok = unsafe {
        windows_sys::Win32::Devices::HumanInterfaceDevice::HidD_SetFeature(
            dev.handle_raw(),
            send_buf.as_ptr() as _,
            feat_size as u32,
        )
    };
    if ok == 0 {
        unsafe extern "system" { fn GetLastError() -> u32; }
        let err = unsafe { GetLastError() };
        return Err(format!("HidD_SetFeature failed (error {err:#010X})").into());
    }
    println!("  (sent via HidD_SetFeature, {feat_size} bytes)");

    std::thread::sleep(Duration::from_millis(50));

    // Read response as GET_FEATURE.
    let mut recv_buf = vec![0u8; feat_size];
    recv_buf[0] = 0x00; // report ID

    let ok = unsafe {
        windows_sys::Win32::Devices::HumanInterfaceDevice::HidD_GetFeature(
            dev.handle_raw(),
            recv_buf.as_mut_ptr() as _,
            feat_size as u32,
        )
    };
    if ok == 0 {
        unsafe extern "system" { fn GetLastError() -> u32; }
        let err = unsafe { GetLastError() };
        println!("RX !! HidD_GetFeature failed (error {err:#010X})");
    } else {
        let response = &recv_buf[1..]; // skip report ID
        println!("RX << {}", hex_dump(&response[..response.len().min(32)]));
        if response[0] == 0x55 {
            println!("  Magic 0x55  REAL RESPONSE!");
            if response.len() >= 32 {
                if response.first() == Some(&0x55) {
                    println!("  Magic 0x55 found  real command response!");
                    print_response_breakdown(&response);
                }
            }
        } else if response[0] == 0xAA {
            println!("  Echo (got back our own data)");
        } else {
            println!("  Unknown format: first byte = {:#04X}", response[0]);
        }
    }

    Ok(())
}

#[cfg(windows)]
fn cmd_bulk(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    use ak680max_driver::protocol::win_hid::{find_device_path, WinHidDevice};

    unsafe extern "system" {
        fn GetLastError() -> u32;
    }

    let path = find_device_path(0x0C45, 0x80B2, 0xFF67)?;
    println!("Opening iface 3 (0xFF67): {path}");
    let dev = WinHidDevice::open(&path)?;

    let caps = dev.get_caps()?;
    let out_size = caps.output_report_byte_length as usize; // 4097
    let feat_size = caps.feature_report_byte_length as usize; // 65
    println!("Output: {out_size}B, Feature: {feat_size}B");

    // Build a 4096-byte payload with a read-config command header.
    // First 32 bytes: AA 10 18 00 00 ... (DeviceInfo as test)
    // or: AA E5 00 00 00 00 01 00 ... (lightless-style read)
    let cmd = args.first().map(|s| s.as_str()).unwrap_or("10");
    let cmd_byte = parse_hex_byte(cmd)?;

    let mut payload = vec![0u8; out_size];
    payload[0] = 0x00; // report ID
    payload[1] = 0xAA;
    payload[2] = cmd_byte;
    payload[3] = 0x18;

    println!("Sending {out_size}B output report (cmd={cmd_byte:#04X})...");
    println!("  Header: {:02X?}", &payload[..16]);

    let ok = unsafe {
        windows_sys::Win32::Devices::HumanInterfaceDevice::HidD_SetOutputReport(
            dev.handle_raw(),
            payload.as_ptr() as _,
            out_size as u32,
        )
    };

    if ok == 0 {
        let err = unsafe { GetLastError() };
        println!("HidD_SetOutputReport failed: {err:#010X}");

        // Try WriteFile instead.
        println!("Trying WriteFile...");

        unsafe extern "system" {
            fn WriteFile(
                hFile: isize,
                lpBuffer: *const u8,
                nNumberOfBytesToWrite: u32,
                lpNumberOfBytesWritten: *mut u32,
                lpOverlapped: *const u8,
            ) -> i32;
        }
    } else {
        println!("HidD_SetOutputReport succeeded!");
    }

    // Now read feature report to see if firmware populated it.
    std::thread::sleep(Duration::from_millis(100));

    let mut recv_buf = vec![0u8; feat_size];
    recv_buf[0] = 0x00;
    let ok = unsafe {
        windows_sys::Win32::Devices::HumanInterfaceDevice::HidD_GetFeature(
            dev.handle_raw(),
            recv_buf.as_mut_ptr() as _,
            feat_size as u32,
        )
    };
    if ok != 0 {
        println!("Feature read after bulk: {}", hex_dump(&recv_buf[1..recv_buf.len().min(33)]));
    }

    // Also check iface 2 for any response.
    println!("Checking iface 2...");
    let api = hidapi::HidApi::new()?;
    if let Some(info) = api.device_list().find(|d| {
        d.vendor_id() == 0x0C45 && d.product_id() == 0x80B2 && d.usage_page() == 0xFF68
    }) {
        if let Ok(reader) = info.open_device(&api) {
            let mut buf = [0u8; 512];
            let n = reader.read_timeout(&mut buf, 500)?;
            if n > 0 {
                println!("  iface 2 got {n} bytes: {}", hex_dump(&buf[..n.min(32)]));
            } else {
                println!("  iface 2: nothing");
            }
        }
    }

    Ok(())
}

fn cmd_dump(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    let size_kb: usize = args.first().map(|s| s.parse()).transpose()?.unwrap_or(64);
    let output = args.get(1).map(|s| s.as_str()).unwrap_or("firmware.bin");
    let cmd = args
        .get(2)
        .map(|s| parse_hex_byte(s))
        .transpose()?
        .unwrap_or(0x17);

    let total_bytes = size_kb * 1024;
    let chunks = (total_bytes + 55) / 56;

    println!("Firmware dump via CMD {cmd:#04X}");
    println!("Size: {size_kb}KB ({total_bytes} bytes, {chunks} chunks)");
    println!("Output: {output}");
    println!();

    let (device, transport, name) = open_device()?;
    println!("Device: {name}");
    println!("Transport: {transport}");
    println!();

    let mut data = Vec::with_capacity(total_bytes);
    let start = Instant::now();

    for chunk in 0..chunks {
        let offset = chunk * 56;
        if offset > 0xFFFF {
            println!("Offset overflow at chunk {chunk} -- stopping at 64KB");
            break;
        }

        let mut report = [0u8; WIRE_REPORT_SIZE];
        report[0] = WIRE_MAGIC_OUT;
        report[1] = cmd;
        report[2] = 0x38; // 56 bytes
        report[3] = (offset & 0xFF) as u8;
        report[4] = ((offset >> 8) & 0xFF) as u8;

        rgb_commands::drain(&device, transport);

        match rgb_commands::probe_command(&device, &report, transport, SEND_TIMEOUT_MS) {
            Ok(Some(response)) => {
                let payload = &response[8..]; // 56 bytes
                data.extend_from_slice(payload);
            }
            Ok(None) => {
                println!("Timeout at offset {offset:#06X} -- padding with FF");
                data.extend_from_slice(&[0xFF; 56]);
            }
            Err(e) => {
                if is_disconnect(&e) {
                    println!("Device disconnected at offset {offset:#06X}");
                    break;
                }
                println!("Error at offset {offset:#06X}: {e} -- padding with FF");
                data.extend_from_slice(&[0xFF; 56]);
            }
        }

        // Progress every 64 chunks (~3.5KB).
        if chunk % 64 == 0 || chunk == chunks - 1 {
            let pct = (chunk + 1) as f64 / chunks as f64 * 100.0;
            let elapsed = start.elapsed().as_secs_f64();
            let speed = data.len() as f64 / elapsed / 1024.0;
            print!(
                "\r  [{:5.1}%] {:#06X} / {:#06X} ({:.1} KB/s)  ",
                pct,
                data.len(),
                total_bytes,
                speed,
            );
            std::io::stdout().flush().ok();
        }
    }

    data.truncate(total_bytes);
    let elapsed = start.elapsed();

    println!();
    println!();
    println!("Dumped {} bytes in {:.1}s", data.len(), elapsed.as_secs_f64());

    // Write to file.
    std::fs::write(output, &data)?;
    println!("Saved to {output}");

    // Quick analysis.
    println!();
    println!("Quick analysis:");

    let non_zero = data.iter().filter(|&&b| b != 0).count();
    let non_ff = data.iter().filter(|&&b| b != 0xFF).count();
    println!("  Non-zero bytes: {non_zero} ({:.1}%)", non_zero as f64 / data.len() as f64 * 100.0);
    println!("  Non-0xFF bytes: {non_ff} ({:.1}%)", non_ff as f64 / data.len() as f64 * 100.0);

    // Look for strings.
    println!();
    println!("Strings found:");
    let mut i = 0;
    while i < data.len() {
        if data[i].is_ascii_graphic() || data[i] == b' ' {
            let start_pos = i;
            while i < data.len() && (data[i].is_ascii_graphic() || data[i] == b' ') {
                i += 1;
            }
            let len = i - start_pos;
            if len >= 4 {
                let s: String = data[start_pos..i].iter().map(|&b| b as char).collect();
                println!("  {start_pos:#06X}: \"{s}\"");
            }
        } else {
            i += 1;
        }
    }

    // Look for ARM vector table (typical at offset 0 of flash).
    // The actuation table is at some offset from flash base.
    // Data at 0x0400 was ARM code, so flash base relative to table start
    // is at -(table_offset). We can detect the initial SP and reset vector.
    println!();
    println!("ARM analysis:");
    for offset in (0..data.len().min(0x200)).step_by(4) {
        let word = u32::from_le_bytes([
            data[offset],
            data.get(offset + 1).copied().unwrap_or(0),
            data.get(offset + 2).copied().unwrap_or(0),
            data.get(offset + 3).copied().unwrap_or(0),
        ]);
        if word >= 0x2000_0000 && word <= 0x2000_FFFF {
            println!("  {offset:#06X}: {word:#010X} (possible SRAM address / initial SP)");
        }
        if word >= 0x0000_0100 && word <= 0x0001_0000 && word & 1 == 1 {
            println!("  {offset:#06X}: {word:#010X} (possible Thumb entry point)");
        }
    }

    Ok(())
}

fn cmd_raw_scan(args: &[String]) -> Result<(), Box<dyn std::error::Error>> {
    use ak680max_driver::protocol::win_hid::{find_device_path, WinHidDevice};

    let start = args.first().map(|s| parse_hex_byte(s)).transpose()?.unwrap_or(0x00);
    let end = args.get(1).map(|s| parse_hex_byte(s)).transpose()?.unwrap_or(0x30);

    let path = find_device_path(0x0C45, 0x80B2, 0xFF67)?;
    println!("Raw scan via HidD_SetOutputReport on iface 3 (0xFF67)");
    println!("Range: {start:#04X}..={end:#04X}");
    let dev = WinHidDevice::open(&path)?;
    println!();

    let mut data_cmds = Vec::new();
    let mut responded = 0u32;

    for cmd in start..=end {
        let mut report = [0u8; 64];
        report[0] = 0xAA;
        report[1] = cmd;
        report[2] = 24; // chunk size

        match dev.set_output_report(&report) {
            Ok(()) => {}
            Err(e) => {
                println!("[{cmd:#04X}] !! send: {e}");
                continue;
            }
        }

        std::thread::sleep(Duration::from_millis(30));

        match dev.get_input_report() {
            Ok(response) => {
                if response[0] == 0x55 && response[1] == cmd {
                    responded += 1;
                    let has_d = response[8..32].iter().any(|&b| b != 0);
                    if has_d {
                        println!("[{cmd:#04X}] << {} ***DATA***", hex_dump(&response[..32]));
                        data_cmds.push(cmd);
                    } else {
                        print!(".");
                        std::io::stdout().flush().ok();
                    }
                } else if response[0] == 0xAA {
                    print!("e"); // echo (not processed)
                    std::io::stdout().flush().ok();
                } else {
                    print!("?");
                    std::io::stdout().flush().ok();
                }
            }
            Err(_) => {
                print!("_");
                std::io::stdout().flush().ok();
            }
        }
    }

    println!();
    println!();
    println!("{responded} responded, {} with data", data_cmds.len());
    for cmd in &data_cmds {
        println!("  {cmd:#04X}");
    }

    Ok(())
}

fn scan_ajazz_devices(api: &HidApi) -> Vec<ScannedDevice> {
    let vid_set: HashSet<u16> = AJAZZ_VENDOR_IDS.iter().copied().collect();
    let total = api.device_list().count();
    log::info!("HID bus: {total} total interface(s) enumerated");

    let mut results = Vec::new();
    for info in api.device_list() {
        if !vid_set.contains(&info.vendor_id()) {
            continue;
        }
        let matched = KNOWN_KEYBOARDS.iter().find(|k| {
            k.vendor_id == info.vendor_id()
                && k.product_id == info.product_id()
                && k.usage_page == info.usage_page()
        });
        let product = info.product_string().unwrap_or("(unknown)").to_string();
        let entry = ScannedDevice {
            vendor_id: info.vendor_id(),
            product_id: info.product_id(),
            usage_page: info.usage_page(),
            usage: info.usage(),
            interface_number: info.interface_number(),
            product_string: product,
            matched_config: matched.map(|k| k.name),
            matched_protocol: matched.map(|k| k.protocol),
        };
        log::info!("  Ajazz HID: {entry}");
        results.push(entry);
    }
    if results.is_empty() {
        log::warn!("No Ajazz-vendor HID interfaces found on bus");
    }
    results
}

pub fn connect() -> Result<(HidDevice, KeyboardState), DeviceError> {
    let api = HidApi::new()?;
    let scanned = scan_ajazz_devices(&api);

    // Try lightless first (full support), then RGB (limited).
    for protocol in [DriverProtocol::Lightless, DriverProtocol::Rgb] {
        if let Some(result) = try_connect_protocol(&api, protocol) {
            return result;
        }
    }

    Err(build_not_found_error(&scanned))
}

fn try_connect_protocol(
    api: &HidApi,
    protocol: DriverProtocol,
) -> Option<Result<(HidDevice, KeyboardState), DeviceError>> {
    for config in KNOWN_KEYBOARDS.iter().filter(|k| k.protocol == protocol) {
        match protocol {
            DriverProtocol::Lightless => {
                // Lightless: match exact usage page.
                for info in api.device_list() {
                    if info.vendor_id() != config.vendor_id
                        || info.product_id() != config.product_id
                        || info.usage_page() != config.usage_page
                    {
                        continue;
                    }
                    log::info!(
                        "Opening {} (Lightless) at {} ...",
                        config.name,
                        info.path().to_string_lossy(),
                    );
                    let device = match info.open_device(api) {
                        Ok(d) => d,
                        Err(e) => {
                            log::error!("open_device failed: {e}");
                            continue;
                        }
                    };
                    return Some(connect_lightless(device, config));
                }
            }
            DriverProtocol::Rgb => {
                // RGB: try every vendor-specific interface (page >= 0xFF00)
                // because the correct interface varies by OS and driver.
                if let Some(result) = try_rgb_interfaces(api, config) {
                    return Some(result);
                }
            }
        }
    }
    None
}

/// Try every vendor-specific HID interface for an RGB keyboard.
///
/// Some RGB models expose multiple vendor-specific interfaces (e.g.
/// 0xFF67 on interface 3 and 0xFF68 on interface 2).  Only one of
/// them may support bidirectional communication.  We probe each in
/// turn until one of them responds to the GetDeviceInfo command.
fn try_rgb_interfaces(
    api: &HidApi,
    config: &KeyboardConfig,
) -> Option<Result<(HidDevice, KeyboardState), DeviceError>> {
    // Collect all vendor-specific interfaces for this VID/PID.
    let mut candidates: Vec<_> = api
        .device_list()
        .filter(|info| {
            info.vendor_id() == config.vendor_id
                && info.product_id() == config.product_id
                && info.usage_page() >= 0xFF00
        })
        .collect();

    if candidates.is_empty() {
        return None;
    }

    // Sort by interface number so we try them in a deterministic order.
    candidates.sort_by_key(|info| info.interface_number());

    let mut any_opened = false;

    for info in &candidates {
        log::info!(
            "Trying RGB: page={:#06X} usage={:#06X} iface={} at {} ...",
            info.usage_page(),
            info.usage(),
            info.interface_number(),
            info.path().to_string_lossy(),
        );

        let device = match info.open_device(api) {
            Ok(d) => d,
            Err(e) => {
                log::warn!("  Could not open: {e}");
                continue;
            }
        };

        any_opened = true;

        match connect_rgb(device, config) {
            Ok(result) => return Some(Ok(result)),
            Err(e) => {
                log::warn!("  Probe failed on this interface: {e}");
                continue;
            }
        }
    }

    if any_opened {
        // We opened at least one interface but none worked.
        Some(Err(DeviceError::Protocol(format!(
            "Tried {} vendor-specific HID interface(s) for {}, none responded",
            candidates.len(),
            config.name,
        ))))
    } else {
        None
    }
}

fn connect_lightless(
    device: HidDevice,
    config: &KeyboardConfig,
) -> Result<(HidDevice, KeyboardState), DeviceError> {
    let key_list = ak680_max_lightless_key_list();

}

pub fn apply_rgb_keys(
    device: &HidDevice,
    keys: &[Option<crate::model::key::Key>],
    transport: rgb_commands::Transport,
    press_table: &[u8],
    release_table: &[u8],
    rgb_table: &[u8],
) -> Result<(Vec<u8>, Vec<u8>, Vec<u8>), DeviceError> {
    let mut new_press = press_table.to_vec();
    let mut new_release = release_table.to_vec();
    let mut new_rgb = rgb_table.to_vec();
    if new_press.len() < 1024 { new_press.resize(1024, 0); }
    if new_release.len() < 1024 { new_release.resize(1024, 0); }
    if new_rgb.len() < 512 { new_rgb.resize(512, 0); }

    for key in keys.iter().flatten() {
        let press_val = (key.down_actuation * 100.0).round() as u16;
        let release_val = (key.up_actuation * 100.0).round() as u16;
        rgb_commands::set_key_actuation(&mut new_press, key.code, press_val);
        rgb_commands::set_key_actuation(&mut new_release, key.code, release_val);
        rgb_commands::set_key_color(
            &mut new_rgb,
            key.code,
            rgb_commands::KeyColor::new(key.color.r, key.color.g, key.color.b),
        );
    }

    rgb_commands::write_actuation_table(device, transport, &new_press)?;
    rgb_commands::write_release_table(device, transport, &new_release)?;
    rgb_commands::write_rgb_table(device, transport, &new_rgb)?;

    Ok((new_press, new_release, new_rgb))
}

fn connect_rgb(
    device: HidDevice,
    config: &KeyboardConfig,
) -> Result<(HidDevice, KeyboardState), DeviceError> {
    log::info!("Probing RGB device transport and querying info...");

    let (raw_info, transport) = rgb_commands::get_device_info(&device)?;

    let device_info = DeviceInfo {
        firmware_version: raw_info.firmware_version,
        battery_level: raw_info.battery_level,
        charge_status: raw_info.charge_status,
        rt_precision: raw_info.rt_precision,
    };

    log::info!(
        "RGB device: firmware={:.2}, battery={}%, charge={}, rt_precision={}, transport={transport}",
        device_info.firmware_version,
        device_info.battery_level,
        device_info.charge_status,
        device_info.rt_precision,
    );

    // Read the actuation table and populate key data.
    log::info!("Reading press actuation table...");
    let press_table = rgb_commands::read_actuation_table(&device, transport)?;

    log::info!("Reading release actuation table...");
    let release_table = rgb_commands::read_release_table(&device, transport)?;

    log::info!("Reading RGB color table...");
    let rgb_table = rgb_commands::read_rgb_table(&device, transport)?;

    let key_list = crate::protocol::key_list::ak680_max_key_list();
    let mut keys: Vec<Option<Key>> = Vec::with_capacity(key_list.len());

    // Firmware returns 0x0000 for "use default". Substitute the actual
    // default so the UI always shows real values. When writing back,
    // this writes 120 (0x0078) explicitly — same effective behaviour.
    const DEFAULT_ACTUATION_MM: f64 = 1.20;

    for (idx, entry) in key_list.iter().enumerate() {
        if entry.is_some() {
            let press_raw = rgb_commands::get_key_actuation(&press_table, idx);
            let release_raw = rgb_commands::get_key_actuation(&release_table, idx);
            let rgb_color = rgb_commands::get_key_color(&rgb_table, idx);

            let press_mm = if press_raw == 0 { DEFAULT_ACTUATION_MM } else { press_raw as f64 / 100.0 };
            let release_mm = if release_raw == 0 { DEFAULT_ACTUATION_MM } else { release_raw as f64 / 100.0 };

            keys.push(Some(Key {
                code: idx,
                down_actuation: press_mm,
                up_actuation: release_mm,
                rapid_trigger: false,
                rt_press_sensitivity: 0.0,
                rt_release_sensitivity: 0.0,
                color: crate::model::key::KeyColor {
                    r: rgb_color.r,
                    g: rgb_color.g,
                    b: rgb_color.b,
                },
            }));
        } else {
            keys.push(None);
        }
    }

    log::info!(
        "Loaded {} keys ({} with non-zero actuation)",
        keys.iter().filter(|k: &&Option<Key>| k.is_some()).count(),
        keys.iter()
            .filter(|k: &&Option<Key>| k.as_ref().map_or(false, |k| k.down_actuation > 0.0))
            .count(),
    );

    let mut state = KeyboardState::new(config.clone(), Layer::Layer1, keys);
    state.device_info = Some(device_info);
    state.transport = Some(transport);
    state.raw_actuation_table = Some(press_table);
    state.raw_release_table = Some(release_table);
    state.raw_rgb_table = Some(rgb_table);
    state.fully_supported = true;

    Ok((device, state))
}

// -
// Commands  high-level HID transactions for the AK680 MAX (lightless).
// -
// Each public function in this module either reads from or writes to the
// keyboard through HID feature reports.  The `hidapi::HidDevice` handle
// must already be open.
// -

use hidapi::HidDevice;

use crate::model::key::Key;
use crate::model::layer::Layer;
use crate::protocol::packet::{build_header, build_packet, REPORT_SIZE};

/// How many chunks the firmware uses for actuation / RT sensitivity data.
const ACTUATION_CHUNK_COUNT: usize = 4;

//  helpers -

/// Direction marker used in actuation and RT sensitivity packets.
#[derive(Clone, Copy)]
enum Direction {
    Press,
    Release,
}

impl Direction {
    /// Byte used in the actuation sub-command field.
    fn actuation_byte(self) -> u8 {
        match self {
            Direction::Press => 0x00,
            Direction::Release => 0x01,
        }
    }

    /// Byte used in the RT sensitivity sub-command field.
    fn rt_sensitivity_byte(self) -> u8 {
        match self {
            Direction::Press => 0x02,
            Direction::Release => 0x03,
        }
    }
}

/// Send a feature report and log it.
fn send(device: &HidDevice, payload: &[u8; REPORT_SIZE]) -> Result<(), hidapi::HidError> {
    // HID feature reports on Windows must be prefixed with the report ID (0).
    let mut buf = [0u8; REPORT_SIZE + 1];
    buf[0] = 0x00;
    buf[1..].copy_from_slice(payload);
    device.send_feature_report(&buf)?;
    log::trace!("Sent feature report: {:02X?}", &payload[..8]);
    Ok(())
}

/// Receive a feature report and return the raw DataView-equivalent buffer.
fn receive(device: &HidDevice) -> Result<[u8; REPORT_SIZE], hidapi::HidError> {
    let mut buf = [0u8; REPORT_SIZE + 1];
    buf[0] = 0x00;
    device.get_feature_report(&mut buf)?;
    let mut out = [0u8; REPORT_SIZE];
    out.copy_from_slice(&buf[1..REPORT_SIZE + 1]);
    log::trace!("Received feature report: {:02X?}", &out[..8]);
    Ok(out)
}

//  public API 

/// Query which layer the keyboard currently has active.
pub fn get_active_layer(device: &HidDevice) -> Result<Layer, crate::device::DeviceError> {
    let header = build_header(0x84, [0; 6]);
    let payload = build_packet(&header, &[]);
    send(device, &payload)?;

    let response = receive(device)?;
    Layer::from_byte(response[1]).ok_or_else(|| {
        crate::device::DeviceError::Protocol(format!("Invalid layer byte: {}", response[1]))
    })
}

/// Tell the keyboard to switch to a different layer.
pub fn set_active_layer(device: &HidDevice, layer: Layer) -> Result<(), hidapi::HidError> {
    let header = build_header(0x04, [layer as u8, 0, 0, 0, 0, 0]);
    let payload = build_packet(&header, &[]);
    send(device, &payload)
}

/// Read every key's settings from the currently active layer.
///
/// The returned vector is indexed by firmware key code; slots for
/// non-existent keys are `None`.
pub fn get_keys(
    device: &HidDevice,
    key_list: &[Option<&'static str>],
) -> Result<Vec<Option<Key>>, crate::device::DeviceError> {
    let mut keys: Vec<Option<Key>> = Vec::with_capacity(key_list.len());
    for (idx, entry) in key_list.iter().enumerate() {
        if entry.is_some() {
            keys.push(Some(Key::new(idx)));
        } else {
            keys.push(None);
        }
    }

    //  rapid trigger on/off (2 chunks of up to 64 bytes) -
    for chunk in 0..2u8 {
        let header = build_header(0xE5, [0x07, 1, chunk, 0, 0, 0]);
        let payload = build_packet(&header, &[]);
        send(device, &payload)?;
        let response = receive(device)?;

        for index in 0..REPORT_SIZE {
            let key_code = chunk as usize * 64 + index;
            if key_code >= keys.len() {
                break;
            }
            if let Some(ref mut key) = keys[key_code] {
                key.rapid_trigger = response[index] == 0x80;
            }
        }
    }

    //  actuation depths and RT sensitivities -
    for direction in [Direction::Press, Direction::Release] {
        for chunk in 0..ACTUATION_CHUNK_COUNT as u8 {
            // Actuation depths
            {
                let header =
                    build_header(0xE5, [direction.actuation_byte(), 1, chunk, 0, 0, 0]);
                let payload = build_packet(&header, &[]);
                send(device, &payload)?;
                let response = receive(device)?;

                // Each actuation value is a little-endian u16 (hundredths of mm).
                for index in 0..(REPORT_SIZE / 2) {
                    let key_code = chunk as usize * 32 + index;
                    if key_code >= keys.len() {
                        break;
                    }
                    let raw = u16::from_le_bytes([
                        response[index * 2],
                        response[index * 2 + 1],
                    ]);
                    if let Some(ref mut key) = keys[key_code] {
                        let mm = raw as f64 / 100.0;
                        match direction {
                            Direction::Press => key.down_actuation = mm,
                            Direction::Release => key.up_actuation = mm,
                        }
                    }
                }
            }

            // RT sensitivity
            {
                let header = build_header(
                    0xE5,
                    [direction.rt_sensitivity_byte(), 1, chunk, 0, 0, 0],
                );
                let payload = build_packet(&header, &[]);
                send(device, &payload)?;
                let response = receive(device)?;

                for index in 0..(REPORT_SIZE / 2) {
                    let key_code = chunk as usize * 32 + index;
                    if key_code >= keys.len() {
                        break;
                    }
                    let raw = u16::from_le_bytes([
                        response[index * 2],
                        response[index * 2 + 1],
                    ]);
                    if let Some(ref mut key) = keys[key_code] {
                        let mm = raw as f64 / 100.0;
                        match direction {
                            Direction::Press => key.rt_press_sensitivity = mm,
                            Direction::Release => key.rt_release_sensitivity = mm,
                        }
                    }
                }
            }
        }
    }

    Ok(keys)
}

/// Write all key settings back to the currently active layer.
pub fn apply_keys(
    device: &HidDevice,
    keys: &[Option<Key>],
) -> Result<(), crate::device::DeviceError> {
    //  rapid trigger on/off (2 chunks of 56 keys each) -
    for chunk in 0..2u8 {
        let start = chunk as usize * 56;
        let end = (start + 56).min(keys.len());
        let slice = &keys[start..end];

        let body: Vec<u8> = slice
            .iter()
            .map(|k| match k {
                Some(key) if key.rapid_trigger => 0x80,
                _ => 0x00,
            })
            .collect();

        let header = build_header(0x65, [0x07, 1, chunk, 0, 0, 0]);
        let payload = build_packet(&header, &body);
        send(device, &payload)?;
    }

    //  actuation depths and RT sensitivities (4 chunks of 28 keys) -
    for direction in [Direction::Press, Direction::Release] {
        for chunk in 0..ACTUATION_CHUNK_COUNT as u8 {
            let start = chunk as usize * 28;
            let end = (start + 28).min(keys.len());
            let slice = &keys[start..end];

            // Actuation
            {
                let body: Vec<u8> = slice
                    .iter()
                    .flat_map(|k| {
                        let val = match k {
                            Some(key) => match direction {
                                Direction::Press => (key.down_actuation * 100.0).round() as u16,
                                Direction::Release => (key.up_actuation * 100.0).round() as u16,
                            },
                            None => 0,
                        };
                        val.to_le_bytes()
                    })
                    .collect();

                let header =
                    build_header(0x65, [direction.actuation_byte(), 1, chunk, 0, 0, 0]);
                let payload = build_packet(&header, &body);
                send(device, &payload)?;
            }

            // Actuation end marker
            {
                let d = direction.actuation_byte();
                let header = build_header(0x65, [d, 0x01, 4, d, 0, 0]);
                let payload = build_packet(&header, &[]);
                send(device, &payload)?;
            }

            // RT sensitivity
            {
                let body: Vec<u8> = slice
                    .iter()
                    .flat_map(|k| {
                        let val = match k {
                            Some(key) => match direction {
                                Direction::Press => {
                                    (key.rt_press_sensitivity * 100.0).round() as u16
                                }
                                Direction::Release => {
                                    (key.rt_release_sensitivity * 100.0).round() as u16
                                }
                            },
                            None => 0,
                        };
                        val.to_le_bytes()
                    })
                    .collect();

                let header = build_header(
                    0x65,
                    [direction.rt_sensitivity_byte(), 1, chunk, 0, 0, 0],
                );
                let payload = build_packet(&header, &body);
                send(device, &payload)?;
            }
        }
    }

    Ok(())
}

// -
// Packet  low-level construction of 64-byte HID feature reports.
// -

/// Size of every HID report exchanged with the keyboard.
pub const REPORT_SIZE: usize = 64;

/// Number of header bytes at the start of each report.
pub const HEADER_SIZE: usize = 8;

/// Build the 8-byte header for a keyboard command.
///
/// `packet_type` is the command identifier.  `data` supplies up to six
/// additional header bytes; the eighth byte is a one-byte checksum computed
/// as `0xFF - (packet_type + sum(data))`.
pub fn build_header(packet_type: u8, data: [u8; 6]) -> [u8; HEADER_SIZE] {
    let sum: u16 = packet_type as u16 + data.iter().map(|&b| b as u16).sum::<u16>();
    let checksum = (0xFFu16.wrapping_sub(sum)) as u8;
    [
        packet_type,
        data[0],
        data[1],
        data[2],
        data[3],
        data[4],
        data[5],
        checksum,
    ]
}

/// Assemble a full 64-byte report from a header and optional body payload.
pub fn build_packet(header: &[u8; HEADER_SIZE], body: &[u8]) -> [u8; REPORT_SIZE] {
    let mut buf = [0u8; REPORT_SIZE];
    buf[..HEADER_SIZE].copy_from_slice(header);
    let copy_len = body.len().min(REPORT_SIZE - HEADER_SIZE);
    buf[HEADER_SIZE..HEADER_SIZE + copy_len].copy_from_slice(&body[..copy_len]);
    buf
}

// -
// RGB commands  HID transport for the AK680 MAX RGB variant.
// -
// The RGB model communicates over 32-byte reports.  Every outgoing packet
// starts with magic byte 0xAA; every incoming response starts with 0x55.
//
// On Windows, some HID interfaces lack an interrupt OUT endpoint, so
// `WriteFile` (used by hidapi's `write()`) hangs with ERROR_IO_PENDING.
// We work around this by probing the device at connection time and falling
// back through three transport strategies:
//
//   1. Output reports  -> `write()` + `read_timeout()`
//   2. Feature reports -> `send_feature_report()` + `get_feature_report()`
//   3. Mixed           -> `send_feature_report()` + `read_timeout()`
//
// The first strategy that successfully completes a round-trip is cached
// and reused for all subsequent communication.
// -

use std::fmt;

use hidapi::HidDevice;

use crate::device::DeviceError;

//  wire constants -

/// Full HID report size for the RGB model (64 data bytes).
const REPORT_SIZE: usize = 64;
const HEADER_SIZE: usize = 8;
/// Payload per chunk (56 bytes, 0x38).
const DATA_SIZE: usize = REPORT_SIZE - HEADER_SIZE;

const MAGIC_OUT: u8 = 0xAA;
const MAGIC_IN: u8 = 0x55;

const CMD_GET_DEVICE_INFO: u8 = 0x10;

const MAX_RETRIES: u32 = 3;
const READ_TIMEOUT_MS: i32 = 1000;

//  transport strategy -

/// Describes which HID API calls to use for send and receive.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum Transport {
    /// `write()` for send, `read_timeout()` for receive.
    OutputReport,
    /// `send_feature_report()` for send, `get_feature_report()` for receive.
    FeatureReport,
    /// `send_feature_report()` for send, `read_timeout()` for receive.
    MixedFeatureWrite,
}

impl fmt::Display for Transport {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Transport::OutputReport => write!(f, "OutputReport (write + read)"),
            Transport::FeatureReport => write!(f, "FeatureReport (feature send + feature get)"),
            Transport::MixedFeatureWrite => {
                write!(f, "Mixed (feature send + interrupt read)")
            }
        }
    }
}

/// All strategies in the order we try them.
const TRANSPORT_PROBE_ORDER: [Transport; 3] = [
    Transport::OutputReport,
    Transport::MixedFeatureWrite,
    Transport::FeatureReport,
];

//  device info result -

/// Information returned by the GetDeviceInfo command.
#[derive(Debug, Clone)]
pub struct RgbDeviceInfo {
    pub firmware_version: f64,
    pub manufacturer_id: u16,
    pub product_id: u16,
    pub battery_level: u8,
    pub charge_status: u8,
    pub rt_precision: u8,
}

//  low-level I/O 

/// Prepend report ID 0x00 to a 32-byte buffer for hidapi calls.
fn prepend_report_id(report: &[u8; REPORT_SIZE]) -> [u8; REPORT_SIZE + 1] {
    let mut buf = [0u8; REPORT_SIZE + 1];
    buf[0] = 0x00;
    buf[1..].copy_from_slice(report);
    buf
}

fn send_raw(
    device: &HidDevice,
    report: &[u8; REPORT_SIZE],
    transport: Transport,
) -> Result<(), DeviceError> {
    // Buffer: report ID (1 byte) + 64 data bytes = 65 bytes.
    let mut buf = [0u8; REPORT_SIZE + 1];
    buf[0] = 0x00;
    buf[1..].copy_from_slice(report);

    match transport {
        Transport::OutputReport => {
            device.write(&buf)?;
        }
        Transport::FeatureReport | Transport::MixedFeatureWrite => {
            device.send_feature_report(&buf)?;
        }
    }
    log::trace!("TX [{transport}]: {:02X?}", &report[..HEADER_SIZE]);
    Ok(())
}

fn recv_raw(
    device: &HidDevice,
    transport: Transport,
) -> Result<[u8; REPORT_SIZE], DeviceError> {
    let mut buf = [0u8; REPORT_SIZE + 1];
    buf[0] = 0x00;

    let n = match transport {
        Transport::OutputReport | Transport::MixedFeatureWrite => {
            device.read_timeout(&mut buf, READ_TIMEOUT_MS)?
        }
        Transport::FeatureReport => device.get_feature_report(&mut buf)?,
    };

    if n == 0 {
        return Err(DeviceError::Protocol(format!(
            "Read returned 0 bytes (transport={transport})"
        )));
    }

    let data_start = if buf[0] == MAGIC_IN {
        0
    } else if n >= 2 && buf[1] == MAGIC_IN {
        1
    } else {
        return Err(DeviceError::Protocol(format!(
            "No magic {MAGIC_IN:#04X} in {n} bytes \
             (first: [{:#04X}, {:#04X}], transport={transport})",
            buf[0],
            buf.get(1).copied().unwrap_or(0),
        )));
    };

    if data_start + REPORT_SIZE > n {
        return Err(DeviceError::Protocol(format!(
            "Short read: {n} bytes from offset {data_start} \
             (need {REPORT_SIZE}, transport={transport})"
        )));
    }

    let mut out = [0u8; REPORT_SIZE];
    out.copy_from_slice(&buf[data_start..data_start + REPORT_SIZE]);
    log::trace!("RX [{transport}]: {:02X?}", &out[..HEADER_SIZE]);
    Ok(out)
}

//  transport probing 

/// Build the GetDeviceInfo request (first chunk only  enough for probing).
fn build_probe_request() -> [u8; REPORT_SIZE] {
    let mut report = [0u8; REPORT_SIZE];
    report[0] = MAGIC_OUT;
    report[1] = CMD_GET_DEVICE_INFO;
    report[2] = 0x30; // length = 48 bytes
    report[5] = 0x01; // sub-command flag
    report
}

/// Try each transport strategy in order by performing a single GetDeviceInfo
/// round-trip.  Returns the first strategy that succeeds together with the
/// response data.
pub fn probe_transport(
    device: &HidDevice,
) -> Result<(Transport, [u8; REPORT_SIZE]), DeviceError> {
    let request = build_probe_request();
    let mut last_error = String::new();

    for &transport in &TRANSPORT_PROBE_ORDER {
        log::info!("Probing transport: {transport} ...");

        match send_raw(device, &request, transport) {
            Ok(()) => {}
            Err(e) => {
                log::warn!("  send failed: {e}");
                last_error = format!("{transport} send: {e}");
                continue;
            }
        }

        match recv_raw(device, transport) {
            Ok(response) => {
                if response[1] == CMD_GET_DEVICE_INFO {
                    log::info!("  Transport {transport} works!");
                    return Ok((transport, response));
                }
                log::warn!(
                    "  Got response but command mismatch: expected {CMD_GET_DEVICE_INFO:#04X}, got {:#04X}",
                    response[1]
                );
                last_error = format!("{transport}: command mismatch");
            }
            Err(e) => {
                log::warn!("  recv failed: {e}");
                last_error = format!("{transport} recv: {e}");
            }
        }
    }

    Err(DeviceError::Protocol(format!(
        "All transport strategies failed. Last: {last_error}"
    )))
}

//  reliable exchange 

/// Send a single report and wait for a command-matching response, with retries.
fn exchange(
    device: &HidDevice,
    report: &[u8; REPORT_SIZE],
    command_type: u8,
    transport: Transport,
) -> Result<[u8; REPORT_SIZE], DeviceError> {
    for attempt in 0..=MAX_RETRIES {
        send_raw(device, report, transport)?;

        match recv_raw(device, transport) {
            Ok(response) => {
                if response[1] == command_type {
                    return Ok(response);
                }
                log::warn!(
                    "Command mismatch: expected {command_type:#04X}, got {:#04X} (attempt {}/{})",
                    response[1],
                    attempt + 1,
                    MAX_RETRIES + 1,
                );
            }
            Err(e) => {
                if attempt < MAX_RETRIES {
                    log::warn!("Read error (attempt {}/{}): {e}", attempt + 1, MAX_RETRIES + 1);
                } else {
                    return Err(e);
                }
            }
        }
    }
    Err(DeviceError::Protocol("All retries exhausted".into()))
}

//  chunked command 

/// Send a multi-chunk command and collect concatenated response data.
pub fn send_command(
    device: &HidDevice,
    command_type: u8,
    data: &[u8],
    header_data: &[u8],
    transport: Transport,
) -> Result<Vec<u8>, DeviceError> {
    if header_data.len() > 3 {
        return Err(DeviceError::Protocol(
            "Header data may not exceed 3 bytes".into(),
        ));
    }

    let chunk_count = if data.is_empty() {
        1
    } else {
        (data.len() + DATA_SIZE - 1) / DATA_SIZE
    };
    let mut response_data = Vec::with_capacity(chunk_count * DATA_SIZE);

    for chunk in 0..chunk_count {
        let offset = chunk * DATA_SIZE;
        let bytes_left = data.len().saturating_sub(offset);
        let chunk_size = if chunk == chunk_count - 1 {
            bytes_left
        } else {
            DATA_SIZE
        };

        let mut report = [0u8; REPORT_SIZE];
        report[0] = MAGIC_OUT;
        report[1] = command_type;
        report[2] = chunk_size as u8;
        report[3] = (offset & 0xFF) as u8;
        report[4] = ((offset >> 8) & 0xFF) as u8;
        for (i, &b) in header_data.iter().enumerate() {
            report[5 + i] = b;
        }
        if chunk_size > 0 && offset < data.len() {
            let end = (offset + chunk_size).min(data.len());
            report[HEADER_SIZE..HEADER_SIZE + (end - offset)]
                .copy_from_slice(&data[offset..end]);
        }

        let response = exchange(device, &report, command_type, transport)?;
        response_data.extend_from_slice(&response[HEADER_SIZE..]);
    }

    Ok(response_data)
}

//  public commands 

/// Query device information, probing for a working transport strategy.
///
/// Returns the device info AND the transport that worked.
pub fn get_device_info(
    device: &HidDevice,
) -> Result<(RgbDeviceInfo, Transport), DeviceError> {
    // DeviceInfo uses length=0x30(48), sub=0x01, single chunk.
    let (transport, response) = probe_transport(device)?;

    let data = &response[HEADER_SIZE..];

    if data.len() < 30 {
        return Err(DeviceError::Protocol(format!(
            "DeviceInfo response too short: {} bytes",
            data.len(),
        )));
    }

    // Bytes [3..5] of data contain version info at offsets 8,9 relative to
    // the original WebHID code (which concatenated multiple 24-byte chunks).
    // In our 56-byte single-chunk response, the layout is the same starting
    // from data[0].
    let firmware_version = {
        let lo = (data[3] & 0x0F) as f64;
        let hi = ((data[3] & 0xF0) >> 4) as f64 * 10.0;
        let major = data[4] as f64 * 100.0;
        (lo + hi + major) / 100.0
    };

    let info = RgbDeviceInfo {
        firmware_version,
        manufacturer_id: u16::from_le_bytes([data[5], data[6]]),
        product_id: u16::from_le_bytes([data[7], data[8]]),
        battery_level: data[10],
        charge_status: data[11],
        rt_precision: data[22],
    };

    log::info!("RGB DeviceInfo: {info:?}");
    log::info!("Working transport: {transport}");
    Ok((info, transport))
}

/// Number of bytes per key record in the actuation table.
const BYTES_PER_KEY: usize = 8;

/// Offset of the u16 actuation value within each key's 8-byte record.
const ACTUATION_FIELD_OFFSET: usize = 2;

/// Total table size in bytes.
const TABLE_SIZE: usize = 1024;

/// Read actuation command.
const CMD_READ_ACTUATION: u8 = 0x17;

/// Write actuation command.
const CMD_WRITE_ACTUATION: u8 = 0x27;

/// Number of full 56-byte chunks needed to cover the table.
const TABLE_CHUNK_COUNT: usize = (TABLE_SIZE + DATA_SIZE - 1) / DATA_SIZE; // 19

// ---------------------------------------------------------------------------
// Per-key RGB color commands.
// ---------------------------------------------------------------------------

const CMD_READ_RGB: u8 = 0x14;
const CMD_WRITE_RGB: u8 = 0x24;

/// Bytes per key in the RGB table: [led_index, R, G, B].
const RGB_BYTES_PER_KEY: usize = 4;

/// Total RGB table size (128 keys * 4 bytes).
const RGB_TABLE_SIZE: usize = 512;

const RGB_CHUNK_COUNT: usize = (RGB_TABLE_SIZE + DATA_SIZE - 1) / DATA_SIZE;

/// Per-key color.
#[derive(Debug, Clone, Copy, Default)]
pub struct KeyColor {
    pub r: u8,
    pub g: u8,
    pub b: u8,
}

impl KeyColor {
    pub fn new(r: u8, g: u8, b: u8) -> Self {
        Self { r, g, b }
    }

    pub fn is_zero(self) -> bool {
        self.r == 0 && self.g == 0 && self.b == 0
    }
}

/// Read the per-key RGB color table.
pub fn read_rgb_table(
    device: &HidDevice,
    transport: Transport,
) -> Result<Vec<u8>, DeviceError> {
    let mut table = Vec::with_capacity(RGB_TABLE_SIZE);

    for chunk in 0..RGB_CHUNK_COUNT {
        let offset = chunk * DATA_SIZE;
        let remaining = RGB_TABLE_SIZE.saturating_sub(offset);
        let chunk_len = remaining.min(DATA_SIZE);

        let mut report = [0u8; REPORT_SIZE];
        report[0] = MAGIC_OUT;
        report[1] = CMD_READ_RGB;
        report[2] = chunk_len as u8;
        report[3] = (offset & 0xFF) as u8;
        report[4] = ((offset >> 8) & 0xFF) as u8;

        let response = exchange(device, &report, CMD_READ_RGB, transport)?;
        table.extend_from_slice(&response[HEADER_SIZE..HEADER_SIZE + chunk_len]);
    }

    table.truncate(RGB_TABLE_SIZE);
    Ok(table)
}

/// Write the per-key RGB color table.
pub fn write_rgb_table(
    device: &HidDevice,
    transport: Transport,
    table: &[u8],
) -> Result<(), DeviceError> {
    if table.len() < RGB_TABLE_SIZE {
        return Err(DeviceError::Protocol(format!(
            "RGB table too short: {} (need {})",
            table.len(),
            RGB_TABLE_SIZE,
        )));
    }

    for chunk in 0..RGB_CHUNK_COUNT {
        let offset = chunk * DATA_SIZE;
        let remaining = RGB_TABLE_SIZE.saturating_sub(offset);
        let chunk_len = remaining.min(DATA_SIZE);
        let is_last = chunk == RGB_CHUNK_COUNT - 1;

        let mut report = [0u8; REPORT_SIZE];
        report[0] = MAGIC_OUT;
        report[1] = CMD_WRITE_RGB;
        report[2] = chunk_len as u8;
        report[3] = (offset & 0xFF) as u8;
        report[4] = ((offset >> 8) & 0xFF) as u8;
        if is_last {
            report[6] = 0x01;
        }
        report[HEADER_SIZE..HEADER_SIZE + chunk_len]
            .copy_from_slice(&table[offset..offset + chunk_len]);

        exchange(device, &report, CMD_WRITE_RGB, transport)?;
    }

    Ok(())
}

/// Get color for a key from the RGB table.
pub fn get_key_color(table: &[u8], key_code: usize) -> KeyColor {
    let offset = key_code * RGB_BYTES_PER_KEY;
    if offset + 4 > table.len() {
        return KeyColor::default();
    }
    // Format: [led_index, R, G, B]
    KeyColor {
        r: table[offset + 1],
        g: table[offset + 2],
        b: table[offset + 3],
    }
}

/// Set color for a key in the RGB table.
pub fn set_key_color(table: &mut [u8], key_code: usize, color: KeyColor) {
    let offset = key_code * RGB_BYTES_PER_KEY;
    if offset + 4 <= table.len() {

/// Read actuation command for release.
const CMD_READ_RELEASE: u8 = 0x18;

/// Write actuation command for release.
const CMD_WRITE_RELEASE: u8 = 0x28;

/// Offset within CMD 0x18 response where release data starts.
const RELEASE_TABLE_OFFSET: usize = 0x400;

/// Total bytes to read for release table (offset + table).
const RELEASE_READ_SIZE: usize = RELEASE_TABLE_OFFSET + TABLE_SIZE;

/// Number of chunks to read the full release area.
const RELEASE_CHUNK_COUNT: usize = (RELEASE_READ_SIZE + DATA_SIZE - 1) / DATA_SIZE;

/// Read the release actuation table from the keyboard.
pub fn read_release_table(
    device: &HidDevice,
    transport: Transport,
) -> Result<Vec<u8>, DeviceError> {
    let mut raw = Vec::with_capacity(RELEASE_READ_SIZE);

    for chunk in 0..RELEASE_CHUNK_COUNT {
        let offset = chunk * DATA_SIZE;
        let remaining = RELEASE_READ_SIZE.saturating_sub(offset);
        let chunk_len = remaining.min(DATA_SIZE);

        let mut report = [0u8; REPORT_SIZE];
        report[0] = MAGIC_OUT;
        report[1] = CMD_READ_RELEASE;
        report[2] = chunk_len as u8;
        report[3] = (offset & 0xFF) as u8;
        report[4] = ((offset >> 8) & 0xFF) as u8;

        let response = exchange(device, &report, CMD_READ_RELEASE, transport)?;
        raw.extend_from_slice(&response[HEADER_SIZE..HEADER_SIZE + chunk_len]);
    }

    if raw.len() < RELEASE_TABLE_OFFSET + TABLE_SIZE {
        raw.resize(RELEASE_TABLE_OFFSET + TABLE_SIZE, 0);
    }
    Ok(raw[RELEASE_TABLE_OFFSET..RELEASE_TABLE_OFFSET + TABLE_SIZE].to_vec())
}

/// Write the release actuation table to the keyboard.
pub fn write_release_table(
    device: &HidDevice,
    transport: Transport,
    table: &[u8],
) -> Result<(), DeviceError> {
    if table.len() < TABLE_SIZE {
        return Err(DeviceError::Protocol(format!(
            "Release table too short: {} (need {})",
            table.len(),
            TABLE_SIZE,
        )));
    }

    let mut full = vec![0u8; RELEASE_TABLE_OFFSET];
    full.extend_from_slice(&table[..TABLE_SIZE]);

    for chunk in 0..RELEASE_CHUNK_COUNT {
        let offset = chunk * DATA_SIZE;
        let remaining = full.len().saturating_sub(offset);
        let chunk_len = remaining.min(DATA_SIZE);
        let is_last = chunk == RELEASE_CHUNK_COUNT - 1;

        let mut report = [0u8; REPORT_SIZE];
        report[0] = MAGIC_OUT;
        report[1] = CMD_WRITE_RELEASE;
        report[2] = chunk_len as u8;
        report[3] = (offset & 0xFF) as u8;
        report[4] = ((offset >> 8) & 0xFF) as u8;
        if is_last {
            report[6] = 0x01;
        }
        report[HEADER_SIZE..HEADER_SIZE + chunk_len]
            .copy_from_slice(&full[offset..offset + chunk_len]);

        exchange(device, &report, CMD_WRITE_RELEASE, transport)?;
    }

    Ok(())
}

/// Read the full 1024-byte actuation table from the keyboard.
pub fn read_actuation_table(
    device: &HidDevice,
    transport: Transport,
) -> Result<Vec<u8>, DeviceError> {

    for chunk in 0..TABLE_CHUNK_COUNT {
        let offset = chunk * DATA_SIZE;
        let remaining = TABLE_SIZE.saturating_sub(offset);
        let chunk_len = remaining.min(DATA_SIZE);

/// Extract the press actuation (hundredths of mm) for a key from the table.
pub fn get_key_actuation(table: &[u8], key_code: usize) -> u16 {
    let offset = key_code * BYTES_PER_KEY + ACTUATION_FIELD_OFFSET;
    if offset + 2 > table.len() {
        return 0;
    }
    u16::from_le_bytes([table[offset], table[offset + 1]])
}

/// Set the press actuation (hundredths of mm) for a key in the table.
pub fn set_key_actuation(table: &mut [u8], key_code: usize, value_hundredths: u16) {
    let offset = key_code * BYTES_PER_KEY + ACTUATION_FIELD_OFFSET;
    if offset + 2 <= table.len() {
        let bytes = value_hundredths.to_le_bytes();
        table[offset] = bytes[0];
        table[offset + 1] = bytes[1];
    }
}

/// Report size exposed for external tools.
pub const WIRE_REPORT_SIZE: usize = REPORT_SIZE;

/// Magic bytes exposed for external tools.
pub const WIRE_MAGIC_OUT: u8 = MAGIC_OUT;
pub const WIRE_MAGIC_IN: u8 = MAGIC_IN;

/// Send a raw 32-byte report and wait for any response.
///
/// Unlike `exchange()`, this does NOT validate the response command byte.
/// Returns `None` on timeout (no data received), `Some(response)` otherwise.
/// Intended for protocol exploration and sniffing.
pub fn probe_command(
    device: &HidDevice,
    report: &[u8; REPORT_SIZE],
    transport: Transport,
    timeout_ms: i32,
) -> Result<Option<[u8; REPORT_SIZE]>, DeviceError> {
    send_raw(device, report, transport)?;

    const BUF_SIZE: usize = 512;
    let mut buf = [0u8; BUF_SIZE];
    buf[0] = 0x00;

    let n = match transport {
        Transport::OutputReport | Transport::MixedFeatureWrite => {
            device.read_timeout(&mut buf, timeout_ms)?
        }
        Transport::FeatureReport => device.get_feature_report(&mut buf)?,
    };

    if n == 0 {
        return Ok(None);
    }

    // Locate magic byte.
    let start = if buf[0] == MAGIC_IN {
        0
    } else if n >= 2 && buf[1] == MAGIC_IN {
        1
    } else {
        return Ok(None);
    };

    if start + REPORT_SIZE > n {
        return Ok(None);
    }

    let mut out = [0u8; REPORT_SIZE];
    out.copy_from_slice(&buf[start..start + REPORT_SIZE]);
    Ok(Some(out))
}

/// Drain any pending input data from the device.
///
/// Reads with a short timeout until nothing comes back, discarding
/// everything.  Useful between probes to clear stale responses.
pub fn drain(device: &HidDevice, transport: Transport) {
    let mut buf = [0u8; 512];
    loop {
        let n = match transport {
            Transport::OutputReport | Transport::MixedFeatureWrite => {
                device.read_timeout(&mut buf, 10).unwrap_or(0)
            }
            Transport::FeatureReport => 0, // Cannot drain feature reports.
        };
        if n == 0 {
            break;
        }
    }
}

unsafe impl Send for WinHidDevice {}
unsafe impl Sync for WinHidDevice {}


impl WinHidDevice {
    /// Open a HID device by its Windows device path.
    pub fn open(path: &str) -> Result<Self, DeviceError> {
        let wide: Vec<u16> = OsStr::new(path).encode_wide().chain(Some(0)).collect();

        let handle = unsafe {
            CreateFileW(
                wide.as_ptr(),
                GENERIC_READ | GENERIC_WRITE,
                FILE_SHARE_READ | FILE_SHARE_WRITE,
                ptr::null(),
                OPEN_EXISTING,
                0,
                ptr::null_mut(),
            )
        };

        if handle == INVALID_HANDLE_VALUE {
            return Err(DeviceError::Protocol(format!(
                "CreateFileW failed for {path}"
            )));
        }

        log::info!("WinHID: opened {path}");
        Ok(Self { handle })
    }
    
    /// Query the HID descriptor for report sizes.
    pub fn get_caps(&self) -> Result<HidPCaps, DeviceError> {
        let mut preparsed: isize = 0;
        let ok = unsafe { HidD_GetPreparsedData(self.handle, &mut preparsed) };
        if ok == 0 {
            return Err(DeviceError::Protocol("HidD_GetPreparsedData failed".into()));
        }

        let mut caps = HidPCaps::default();
        let status = unsafe { HidP_GetCaps(preparsed, &mut caps) };
        unsafe { HidD_FreePreparsedData(preparsed) };

        if status != 0x00110000 {
            // HIDP_STATUS_SUCCESS = 0x00110000
            return Err(DeviceError::Protocol(format!(
                "HidP_GetCaps failed: status {status:#010X}"
            )));
        }

        Ok(caps)
    }

    pub fn handle_raw(&self) -> HANDLE {
        self.handle
    }

    /// Send an OUTPUT report with the exact size the HID descriptor expects.
    pub fn set_output_report_dynamic(&self, data: &[u8], report_size: usize) -> Result<(), DeviceError> {
        let buf_size = report_size; // includes report ID
        let mut buf = vec![0u8; buf_size];
        buf[0] = 0x00; // report ID
        let copy_len = data.len().min(buf_size - 1);
        buf[1..1 + copy_len].copy_from_slice(&data[..copy_len]);

            HidD_SetOutputReport(self.handle, buf.as_ptr() as _, buf_size as u32)
        };

            let err = unsafe { GetLastError() };
            Err(DeviceError::Protocol(format!(

        }

        }
    }

        }

pub fn find_device_path(

    pid: u16,
    usage_page: u16,

    let api = hidapi::HidApi::new()?;
    for info in api.device_list() {

        }
    }
    Err(DeviceError::NotFound(format!(
        "No HID device with VID={vid:#06X} PID={pid:#06X} page={usage_page:#06X}"
    )))
}

struct Snap { down_actuation: f64, up_actuation: f64, rapid_trigger: bool, rt_press: f64, rt_release: f64 }

    Some(Snap { down_actuation: k.down_actuation, up_actuation: k.up_actuation,

// ─── Background worker ────────────────────────────────────────

fn bg_worker(rx: Receiver<BgReq>, tx: Sender<BgResp>, hid: Arc<Mutex<Option<HidDevice>>>) {
    while let Ok(req) = rx.recv() {
        match req {
            BgReq::Connect => match connection::connect() {
                Ok((d, s)) => { let _ = tx.send(BgResp::Connected(d, s)); }
                Err(e) => { let _ = tx.send(BgResp::Error(format!("{e}"))); }
            },
            BgReq::SwitchLayer(l) => {
                let g = hid.lock();
                if let Some(ref d) = *g {
                    match connection::switch_layer(d, l) {
                        Ok(k) => { let _ = tx.send(BgResp::LayerSwitched(l, k)); }
                        Err(e) => { let _ = tx.send(BgResp::Error(format!("{e}"))); }
                    }
                }
            }
            BgReq::Apply(keys) => {
                let g = hid.lock();
                if let Some(ref d) = *g {
                    match connection::apply_all_keys(d, &keys) {
                        Ok(()) => { let _ = tx.send(BgResp::Applied); }
                        Err(e) => { let _ = tx.send(BgResp::Error(format!("{e}"))); }
                    }
                }
            }
            BgReq::ApplyRgb(keys, tr, p, r, c) => {
                let g = hid.lock();
                if let Some(ref d) = *g {
                    match connection::apply_rgb_keys(d, &keys, tr, &p, &r, &c) {
                        Ok((np, nr, nc)) => { let _ = tx.send(BgResp::AppliedRgb(np, nr, nc)); }
                        Err(e) => { let _ = tx.send(BgResp::Error(format!("{e}"))); }
                    }
                }
            }
        }
    }
}

use std::f32::consts::PI;

#[repr(C)]
#[derive(Clone, Copy, Default)]
pub struct Vertex {
    pub pos:   [f32; 2],
    pub uv:    [f32; 2],
    pub color: [f32; 4],
}

pub struct DrawCmd {
    pub index_count:  u32,
    pub index_offset: u32,
    pub clip: [i32; 4],
}

pub struct DrawList {
    pub vertices: Vec<Vertex>,
    pub indices:  Vec<u32>,
    pub commands: Vec<DrawCmd>,
    clip: [i32; 4],
}

const WHITE_UV: [f32; 2] = [0.5 / 512.0, 0.5 / 512.0];

/// Segments per corner of a rounded rect. 12 gives sub-pixel
/// chord error even at 16 px radius — visually indistinguishable
/// from a true arc.
const CORNER_SEG: u32 = 12;

/// Segments for a full circle.  64 is smooth even at r = 80+.
const CIRCLE_SEG: u32 = 64;

impl DrawList {
    pub fn new(w: u32, h: u32) -> Self {
        Self {
            vertices: Vec::with_capacity(48_000),
            indices:  Vec::with_capacity(96_000),
            commands: Vec::new(),
            clip: [0, 0, w as i32, h as i32],
        }
    }

    pub fn clear(&mut self, w: u32, h: u32) {
        self.vertices.clear();
        self.indices.clear();
        self.commands.clear();
        self.clip = [0, 0, w as i32, h as i32];
    }

    pub fn set_clip(&mut self, x: f32, y: f32, w: f32, h: f32) {
        self.clip = [x as i32, y as i32, w as i32, h as i32];
    }

    pub fn reset_clip(&mut self, sw: u32, sh: u32) {
        self.clip = [0, 0, sw as i32, sh as i32];
    }

    fn emit(&mut self, idx_count: u32, idx_offset: u32) {
        if let Some(last) = self.commands.last_mut() {
            if last.clip == self.clip && last.index_offset + last.index_count == idx_offset {
                last.index_count += idx_count;
                return;
            }
        }
        self.commands.push(DrawCmd { index_count: idx_count, index_offset: idx_offset, clip: self.clip });
    }

    // ─── Solid primitives ──────────────────────────────────────

    pub fn rect(&mut self, x: f32, y: f32, w: f32, h: f32, color: [f32; 4]) {
        let b = self.vertices.len() as u32;
        self.vertices.extend_from_slice(&[
            Vertex { pos: [x, y],         uv: WHITE_UV, color },
            Vertex { pos: [x + w, y],     uv: WHITE_UV, color },
            Vertex { pos: [x + w, y + h], uv: WHITE_UV, color },
            Vertex { pos: [x, y + h],     uv: WHITE_UV, color },
        ]);
        let io = self.indices.len() as u32;
        self.indices.extend_from_slice(&[b, b+1, b+2, b, b+2, b+3]);
        self.emit(6, io);
    }

    pub fn rounded_rect(&mut self, x: f32, y: f32, w: f32, h: f32, r: f32, color: [f32; 4]) {
        if r < 0.5 || w < r * 2.0 || h < r * 2.0 {
            return self.rect(x, y, w, h, color);
        }
        let b = self.vertices.len() as u32;
        let seg = CORNER_SEG;
        self.vertices.push(Vertex { pos: [x + w * 0.5, y + h * 0.5], uv: WHITE_UV, color });
        let corners = [
            (x + r,     y + r,     PI,       PI * 1.5),
            (x + w - r, y + r,     PI * 1.5, PI * 2.0),
            (x + w - r, y + h - r, 0.0,      PI * 0.5),
            (x + r,     y + h - r, PI * 0.5, PI),
        ];
        for &(cx, cy, a0, a1) in &corners {
            for i in 0..=seg {
                let a = a0 + (a1 - a0) * i as f32 / seg as f32;
                self.vertices.push(Vertex { pos: [cx + r * a.cos(), cy + r * a.sin()], uv: WHITE_UV, color });
            }
        }
        let n = 4 * (seg + 1);
        let io = self.indices.len() as u32;
        for i in 0..n {
            self.indices.extend_from_slice(&[b, b + 1 + i, b + 1 + (i + 1) % n]);
        }
        self.emit(n * 3, io);
    }

    pub fn circle(&mut self, cx: f32, cy: f32, r: f32, color: [f32; 4]) {
        let b = self.vertices.len() as u32;
        let seg = CIRCLE_SEG;
        self.vertices.push(Vertex { pos: [cx, cy], uv: WHITE_UV, color });
        for i in 0..seg {
            let a = 2.0 * PI * i as f32 / seg as f32;
            self.vertices.push(Vertex { pos: [cx + r * a.cos(), cy + r * a.sin()], uv: WHITE_UV, color });
        }
        let io = self.indices.len() as u32;
        for i in 0..seg {
            self.indices.extend_from_slice(&[b, b + 1 + i, b + 1 + (i + 1) % seg]);
        }
        self.emit(seg * 3, io);
    }

    /// Ring (annulus) — useful for the HSV hue wheel soft edges.
    pub fn ring(&mut self, cx: f32, cy: f32, r_inner: f32, r_outer: f32, seg: u32, color: [f32; 4]) {
        let b = self.vertices.len() as u32;
        for i in 0..seg {
            let a = 2.0 * PI * i as f32 / seg as f32;
            let (s, c_a) = (a.sin(), a.cos());
            self.vertices.push(Vertex { pos: [cx + r_inner * c_a,  cy + r_inner * s],  uv: WHITE_UV, color });
            self.vertices.push(Vertex { pos: [cx + r_outer * c_a, cy + r_outer * s], uv: WHITE_UV, color });
        }
        let io = self.indices.len() as u32;
        let mut count = 0u32;
        for i in 0..seg {
            let i0 = i * 2;
            let i1 = ((i + 1) % seg) * 2;
            self.indices.extend_from_slice(&[b+i0, b+i0+1, b+i1+1, b+i0, b+i1+1, b+i1]);
            count += 6;
        }
        self.emit(count, io);
    }

        ]);

    pub fn ring_segment(

pub const ATLAS_W: u32 = 512;
pub const ATLAS_H: u32 = 512;

struct GlyphEntry { ax: u32, ay: u32, w: u32, h: u32, off_x: f32, off_y: f32, advance: f32 }

pub struct FontAtlas {
    pub pixels: Vec<u8>,
    glyphs: HashMap<(char, u16), GlyphEntry>,
    font: fontdue::Font,
    cursor_x: u32,
    cursor_y: u32,
    row_h: u32,
}

fn size_key(s: f32) -> u16 { (s * 10.0) as u16 }

impl FontAtlas {
    pub fn new() -> Self {
        let paths = [
            "C:\\Windows\\Fonts\\bahnschrift.ttf",
            "C:\\Windows\\Fonts\\segoeui.ttf",
            "C:\\Windows\\Fonts\\arial.ttf",
            "C:\\Windows\\Fonts\\tahoma.ttf",
        ];
        let data = paths.iter().find_map(|p| std::fs::read(p).ok()).expect("No system font found");
        let font = fontdue::Font::from_bytes(data, fontdue::FontSettings::default()).expect("Failed to parse font");

        let mut atlas = Self {
            pixels: vec![0u8; (ATLAS_W * ATLAS_H) as usize],
            glyphs: HashMap::new(),
            font,
            cursor_x: 4, cursor_y: 0, row_h: 0,
        };

        // White pixel block for solid fills.
        for y in 0..3u32 { for x in 0..3u32 { atlas.pixels[(y * ATLAS_W + x) as usize] = 255; } }

        let sizes = [8.0f32, 10.0, 11.0, 12.0, 13.0, 14.0, 15.0, 16.0, 22.0, 24.0];
        for &sz in &sizes {
            for c in 32u8..=126 { atlas.rasterize(c as char, sz); }
            // Unicode icons for buttons.
            for &c in &['\u{2713}', '\u{25C6}', '\u{25B8}', '\u{2022}'] {
                atlas.rasterize(c, sz);
            }
        }
        atlas
    }

    fn rasterize(&mut self, ch: char, size: f32) {
        let key = (ch, size_key(size));
        if self.glyphs.contains_key(&key) { return; }
        let (metrics, bitmap) = self.font.rasterize(ch, size);
        let (w, h) = (metrics.width as u32, metrics.height as u32);
        if w == 0 || h == 0 {
            self.glyphs.insert(key, GlyphEntry { ax: 0, ay: 0, w: 0, h: 0, off_x: metrics.xmin as f32, off_y: 0.0, advance: metrics.advance_width });
            return;
        }
        if self.cursor_x + w + 1 > ATLAS_W { self.cursor_y += self.row_h + 1; self.cursor_x = 0; self.row_h = 0; }
        if self.cursor_y + h + 1 > ATLAS_H { log::warn!("Font atlas full for '{ch}' at {size}"); return; }
        let (ax, ay) = (self.cursor_x, self.cursor_y);
        for row in 0..h {
            let src = (row * w) as usize;
            let dst = ((ay + row) * ATLAS_W + ax) as usize;
            self.pixels[dst..dst + w as usize].copy_from_slice(&bitmap[src..src + w as usize]);
        }
        self.cursor_x += w + 1;
        self.row_h = self.row_h.max(h);
        let ascent = self.font.horizontal_line_metrics(size).map(|l| l.ascent).unwrap_or(size * 0.8);
        let off_y = ascent - metrics.ymin as f32 - h as f32;
        self.glyphs.insert(key, GlyphEntry { ax, ay, w, h, off_x: metrics.xmin as f32, off_y, advance: metrics.advance_width });
    }

    pub fn measure(&self, text: &str, size: f32) -> (f32, f32) {
        let ks = size_key(size);
        let mut w = 0.0f32;
        let h = self.font.horizontal_line_metrics(size).map(|l| l.ascent - l.descent).unwrap_or(size);
        for ch in text.chars() { if let Some(g) = self.glyphs.get(&(ch, ks)) { w += g.advance; } }
        (w, h)
    }

    pub fn draw_text(&self, dl: &mut DrawList, text: &str, x: f32, y: f32, size: f32, color: [f32; 4]) {
        let ks = size_key(size);
        let (aw, ah) = (ATLAS_W as f32, ATLAS_H as f32);
        let mut cx = x;
        for ch in text.chars() {
            if let Some(g) = self.glyphs.get(&(ch, ks)) {
                if g.w > 0 && g.h > 0 {
                    let gx = (cx + g.off_x).round();
                    let gy = (y + g.off_y).round();
                    dl.glyph(gx, gy, g.w as f32, g.h as f32,
                        g.ax as f32 / aw, g.ay as f32 / ah,
                        (g.ax + g.w) as f32 / aw, (g.ay + g.h) as f32 / ah, color);
                }
                cx += g.advance;
            }
        }
    }

    pub fn draw_centered(&self, dl: &mut DrawList, text: &str, cx: f32, cy: f32, size: f32, color: [f32; 4]) {
        let (tw, th) = self.measure(text, size);
        self.draw_text(dl, text, (cx - tw * 0.5).round(), (cy - th * 0.5).round(), size, color);
    }
}

pub fn hsv_to_rgb(h: f32, s: f32, v: f32) -> [f32; 3] {
    let h = ((h % 360.0) + 360.0) % 360.0;
    let c = v * s;
    let hp = h / 60.0;
    let x = c * (1.0 - ((hp % 2.0) - 1.0).abs());
    let (r1, g1, b1) = match hp as u32 {
        0 => (c, x, 0.0), 1 => (x, c, 0.0), 2 => (0.0, c, x),
        3 => (0.0, x, c), 4 => (x, 0.0, c), _ => (c, 0.0, x),
    };
    let m = v - c;
    [r1 + m, g1 + m, b1 + m]
}

pub fn hsv_to_rgba(h: f32, s: f32, v: f32) -> [f32; 4] {
    let [r, g, b] = hsv_to_rgb(h, s, v);
    [r, g, b, 1.0]
}

pub fn rgb_to_hsv(r: f32, g: f32, b: f32) -> (f32, f32, f32) {
    let max = r.max(g).max(b);
    let min = r.min(g).min(b);
    let d = max - min;
    let s = if max < 1e-6 { 0.0 } else { d / max };
    let h = if d < 1e-6 {
        0.0
    } else if (max - r).abs() < 1e-6 {
        60.0 * (((g - b) / d) % 6.0)
    } else if (max - g).abs() < 1e-6 {
        60.0 * ((b - r) / d + 2.0)
    } else {
        60.0 * ((r - g) / d + 4.0)
    };
    (if h < 0.0 { h + 360.0 } else { h }, s, max)
}

    unsafe fn create_swapchain(&mut self, w: u32, h: u32) {
        let caps = self.surface_fn.get_physical_device_surface_capabilities(self.pdev, self.surface).unwrap();
        let extent = vk::Extent2D {
            width:  w.clamp(caps.min_image_extent.width, caps.max_image_extent.width),
            height: h.clamp(caps.min_image_extent.height, caps.max_image_extent.height),
        };
        let img_count = (caps.min_image_count + 1).min(if caps.max_image_count > 0 { caps.max_image_count } else { u32::MAX });
        let sci = vk::SwapchainCreateInfoKHR::default().surface(self.surface).min_image_count(img_count)
            .image_format(vk::Format::B8G8R8A8_UNORM).image_color_space(vk::ColorSpaceKHR::SRGB_NONLINEAR)
            .image_extent(extent).image_array_layers(1).image_usage(vk::ImageUsageFlags::COLOR_ATTACHMENT)
            .image_sharing_mode(vk::SharingMode::EXCLUSIVE).pre_transform(caps.current_transform)
            .composite_alpha(vk::CompositeAlphaFlagsKHR::OPAQUE).present_mode(vk::PresentModeKHR::FIFO)
            .old_swapchain(self.sc);
        let new_sc = self.swap_fn.create_swapchain(&sci, None).unwrap();
        if self.sc != vk::SwapchainKHR::null() { self.destroy_swapchain_views(); self.swap_fn.destroy_swapchain(self.sc, None); }
        self.sc = new_sc;
        self.sc_ext = extent;
        let images = self.swap_fn.get_swapchain_images(self.sc).unwrap();
        self.sc_views = images.iter().map(|&img| {
            let ivci = vk::ImageViewCreateInfo::default().image(img).view_type(vk::ImageViewType::TYPE_2D).format(self.sc_fmt)
                .subresource_range(vk::ImageSubresourceRange { aspect_mask: vk::ImageAspectFlags::COLOR, base_mip_level: 0, level_count: 1, base_array_layer: 0, layer_count: 1 });
            self.dev.create_image_view(&ivci, None).unwrap()
        }).collect();
        self.fbs = self.sc_views.iter().map(|&v| {
            let atts = [v];
            let fbci = vk::FramebufferCreateInfo::default().render_pass(self.rp).attachments(&atts).width(extent.width).height(extent.height).layers(1);
            self.dev.create_framebuffer(&fbci, None).unwrap()
        }).collect();
    }

    unsafe fn destroy_swapchain_views(&mut self) {
        for &fb in &self.fbs { self.dev.destroy_framebuffer(fb, None); }
        for &v in &self.sc_views { self.dev.destroy_image_view(v, None); }
        self.fbs.clear(); self.sc_views.clear();
    }

    pub fn resize(&mut self, w: u32, h: u32) {
        if w == 0 || h == 0 { return; }
        unsafe { self.dev.device_wait_idle().unwrap(); self.create_swapchain(w, h); }
    }

    pub fn upload_atlas(&mut self, data: &[u8]) {
        unsafe {
            std::ptr::copy_nonoverlapping(data.as_ptr(), self.staging.ptr, data.len());
            let cmd = self.one_time_begin();
            let range = vk::ImageSubresourceRange { aspect_mask: vk::ImageAspectFlags::COLOR, base_mip_level: 0, level_count: 1, base_array_layer: 0, layer_count: 1 };
            let bar = [vk::ImageMemoryBarrier::default().image(self.atlas_img).subresource_range(range)
                .old_layout(vk::ImageLayout::UNDEFINED).new_layout(vk::ImageLayout::TRANSFER_DST_OPTIMAL)
                .dst_access_mask(vk::AccessFlags::TRANSFER_WRITE)];
            self.dev.cmd_pipeline_barrier(cmd, vk::PipelineStageFlags::TOP_OF_PIPE, vk::PipelineStageFlags::TRANSFER, vk::DependencyFlags::empty(), &[], &[], &bar);
            let region = [vk::BufferImageCopy {
                image_subresource: vk::ImageSubresourceLayers { aspect_mask: vk::ImageAspectFlags::COLOR, mip_level: 0, base_array_layer: 0, layer_count: 1 },
                image_extent: vk::Extent3D { width: ATLAS_W, height: ATLAS_H, depth: 1 }, ..Default::default()
            }];
            self.dev.cmd_copy_buffer_to_image(cmd, self.staging.buf, self.atlas_img, vk::ImageLayout::TRANSFER_DST_OPTIMAL, &region);
            let bar2 = [vk::ImageMemoryBarrier::default().image(self.atlas_img).subresource_range(range)
                .old_layout(vk::ImageLayout::TRANSFER_DST_OPTIMAL).new_layout(vk::ImageLayout::SHADER_READ_ONLY_OPTIMAL)
                .src_access_mask(vk::AccessFlags::TRANSFER_WRITE).dst_access_mask(vk::AccessFlags::SHADER_READ)];
            self.dev.cmd_pipeline_barrier(cmd, vk::PipelineStageFlags::TRANSFER, vk::PipelineStageFlags::FRAGMENT_SHADER, vk::DependencyFlags::empty(), &[], &[], &bar2);
            self.one_time_end(cmd);
        }
    }

    unsafe fn one_time_begin(&self) -> vk::CommandBuffer {
        let ai = vk::CommandBufferAllocateInfo::default().command_pool(self.pool).level(vk::CommandBufferLevel::PRIMARY).command_buffer_count(1);
        let cmd = self.dev.allocate_command_buffers(&ai).unwrap()[0];
        self.dev.begin_command_buffer(cmd, &vk::CommandBufferBeginInfo::default().flags(vk::CommandBufferUsageFlags::ONE_TIME_SUBMIT)).unwrap();
        cmd
    }

    unsafe fn one_time_end(&self, cmd: vk::CommandBuffer) {
        self.dev.end_command_buffer(cmd).unwrap();
        let si = [vk::SubmitInfo::default().command_buffers(std::slice::from_ref(&cmd))];
        self.dev.queue_submit(self.queue, &si, vk::Fence::null()).unwrap();
        self.dev.queue_wait_idle(self.queue).unwrap();
        self.dev.free_command_buffers(self.pool, &[cmd]);
    }

    pub fn render(&mut self, verts: &[Vertex], idxs: &[u32], cmds: &[DrawCmd]) {
        if self.sc_ext.width == 0 || self.sc_ext.height == 0 { return; }
        unsafe { self.render_inner(verts, idxs, cmds) }
    }

    unsafe fn render_inner(&mut self, verts: &[Vertex], idxs: &[u32], draw_cmds: &[DrawCmd]) {
        let f = self.frame;
        self.dev.wait_for_fences(&[self.fences[f]], true, u64::MAX).unwrap();
        let acq = self.swap_fn.acquire_next_image(self.sc, u64::MAX, self.sem_avail[f], vk::Fence::null());
        let img_idx = match acq {
            Ok((idx, _)) => idx,
            Err(vk::Result::ERROR_OUT_OF_DATE_KHR) => { self.resize(self.sc_ext.width, self.sc_ext.height); return; }
            Err(e) => panic!("acquire: {e}"),
        };
        self.dev.reset_fences(&[self.fences[f]]).unwrap();

        let vb_bytes = std::mem::size_of_val(verts);
        let ib_bytes = std::mem::size_of_val(idxs);
        if vb_bytes > 0 { std::ptr::copy_nonoverlapping(verts.as_ptr() as *const u8, self.vb[f].ptr, vb_bytes); }
        if ib_bytes > 0 { std::ptr::copy_nonoverlapping(idxs.as_ptr()  as *const u8, self.ib[f].ptr, ib_bytes); }

        let cmd = self.cmds[f];
        self.dev.reset_command_buffer(cmd, vk::CommandBufferResetFlags::empty()).unwrap();
        self.dev.begin_command_buffer(cmd, &vk::CommandBufferBeginInfo::default()).unwrap();

        let clear = [vk::ClearValue { color: vk::ClearColorValue { float32: [t::BG_BASE[0], t::BG_BASE[1], t::BG_BASE[2], 1.0] } }];
        let rpbi = vk::RenderPassBeginInfo::default().render_pass(self.rp).framebuffer(self.fbs[img_idx as usize])
            .render_area(vk::Rect2D { offset: vk::Offset2D::default(), extent: self.sc_ext }).clear_values(&clear);
        self.dev.cmd_begin_render_pass(cmd, &rpbi, vk::SubpassContents::INLINE);
        self.dev.cmd_bind_pipeline(cmd, vk::PipelineBindPoint::GRAPHICS, self.pl);

        let vp = [vk::Viewport { x: 0.0, y: 0.0, width: self.sc_ext.width as f32, height: self.sc_ext.height as f32, min_depth: 0.0, max_depth: 1.0 }];
        self.dev.cmd_set_viewport(cmd, 0, &vp);

        let pc = [self.sc_ext.width as f32, self.sc_ext.height as f32];
        self.dev.cmd_push_constants(cmd, self.pl_layout, vk::ShaderStageFlags::VERTEX, 0,
            std::slice::from_raw_parts(pc.as_ptr() as *const u8, 8));

        self.dev.cmd_bind_vertex_buffers(cmd, 0, &[self.vb[f].buf], &[0]);
        self.dev.cmd_bind_index_buffer(cmd, self.ib[f].buf, 0, vk::IndexType::UINT32);
        self.dev.cmd_bind_descriptor_sets(cmd, vk::PipelineBindPoint::GRAPHICS, self.pl_layout, 0, &[self.ds], &[]);

        for dc in draw_cmds {
            let sc = vk::Rect2D { offset: vk::Offset2D { x: dc.clip[0], y: dc.clip[1] },
                extent: vk::Extent2D { width: dc.clip[2] as u32, height: dc.clip[3] as u32 } };
            self.dev.cmd_set_scissor(cmd, 0, &[sc]);
            self.dev.cmd_draw_indexed(cmd, dc.index_count, 1, dc.index_offset, 0, 0);
        }

        self.dev.cmd_end_render_pass(cmd);
        self.dev.end_command_buffer(cmd).unwrap();

        let wait_sem = [self.sem_avail[f]];
        let sig_sem = [self.sem_done[f]];
        let wait_stages = [vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT];
        let si = [vk::SubmitInfo::default().wait_semaphores(&wait_sem).wait_dst_stage_mask(&wait_stages)
            .command_buffers(std::slice::from_ref(&cmd)).signal_semaphores(&sig_sem)];
        self.dev.queue_submit(self.queue, &si, self.fences[f]).unwrap();

        let scs = [self.sc]; let idxs_p = [img_idx];
        let pi = vk::PresentInfoKHR::default().wait_semaphores(&sig_sem).swapchains(&scs).image_indices(&idxs_p);
        let _ = self.swap_fn.queue_present(self.queue, &pi);
        self.frame = (f + 1) % MAX_FRAMES;
    }
}

    let track_h = 5.0;
    let thumb_r = 7.0;

    fa.draw_text(dl, label, x, y + 2.0, t::FONT_SMALL, label_col);
    let lw = fa.measure(label, t::FONT_SMALL).0 + 8.0;
    let sx = x + lw;
    let sw = w - lw - 60.0;
    let ty = y + h * 0.5;

    // Track background with subtle border
    dl.rounded_rect(sx, ty - track_h * 0.5, sw, track_h, track_h * 0.5, t::SLIDER_TRACK);
    dl.rounded_rect(sx, ty - track_h * 0.5, sw, track_h, track_h * 0.5, t::SLIDER_TRACK_BORDER);
    dl.rounded_rect(
        sx + 0.5, ty - track_h * 0.5 + 0.5,
        sw - 1.0, track_h - 1.0,
        (track_h - 1.0) * 0.5, t::SLIDER_TRACK,
    );

    let target = ((*val - min) / (max - min)).clamp(0.0, 1.0) as f32;
    state.visual_frac = input::smooth(state.visual_frac, target, t::ANIM_SLOW, inp.dt);
    let thumb_x = sx + state.visual_frac * sw;

    // Filled portion
    if state.visual_frac > 0.005 {
        dl.rounded_rect(sx, ty - track_h * 0.5, thumb_x - sx, track_h, track_h * 0.5, t::ACCENT);
    }

    // Hover
    let hovered = inp.in_rect(sx - thumb_r, y, sw + thumb_r * 2.0, h);
    state.hover_t = input::smooth(
        state.hover_t,
        if hovered || state.dragging { 1.0 } else { 0.0 },
        t::ANIM_FAST, inp.dt,
    );

    let tr = thumb_r + state.hover_t * 1.5;

    // Focus ring (accent glow) when dragging
    if state.dragging {
        dl.circle(thumb_x, ty, tr + 5.0, t::with_alpha(t::ACCENT, 0.25));
    } else if state.hover_t > 0.01 {
        dl.circle(thumb_x, ty, tr + 3.0, t::with_alpha(t::ACCENT, 0.12 * state.hover_t));
    }

    // Thumb shadow
    dl.circle(thumb_x, ty + 1.0, tr + 0.5, [0.0, 0.0, 0.0, 0.22]);
    // Thumb
    let tc = t::lerp4([0.82, 0.82, 0.82, 1.0], t::TEXT, state.hover_t);
    dl.circle(thumb_x, ty, tr, tc);

    // Value text
    let txt = format!("{:.2} mm", val);
    fa.draw_text(dl, &txt, x + w - 55.0, y + 6.0, t::FONT_SMALL, t::TEXT_SEC);

    // Interaction
    if inp.just_pressed() && hovered { state.dragging = true; }
    if !inp.mouse_down { state.dragging = false; }

    let mut changed = false;
    if state.dragging {
        let frac = ((inp.mouse_x - sx) / sw).clamp(0.0, 1.0) as f64;
        let q = ((min + frac * (max - min)) / step).round() * step;
        if (*val - q).abs() > 1e-9 { *val = q; changed = true; }
    }
    changed
}

#[derive(Default)]
pub struct ButtonAnim { anims: HashMap<u64, f32> }

impl ButtonAnim {
    fn ht(&mut self, x: f32, y: f32, hov: bool, dt: f32) -> f32 {
        let k = ((x as u32 as u64) << 32) | (y as u32 as u64);
        let v = self.anims.entry(k).or_insert(0.0);
        *v = input::smooth(*v, if hov { 1.0 } else { 0.0 }, t::ANIM_FAST, dt);
        *v
    }
}

pub fn button_anim(
    dl: &mut DrawList, fa: &FontAtlas, inp: &InputState,
    anims: &mut ButtonAnim,
    x: f32, y: f32, w: f32, h: f32, label: &str, bg: [f32; 4], enabled: bool,
) -> bool {
    let hov = enabled && inp.in_rect(x, y, w, h);
    let pressed = hov && inp.mouse_down;
    let ht = anims.ht(x, y, hov, inp.dt);
    let col = if pressed { t::lerp4(bg, [0.0, 0.0, 0.0, 1.0], 0.25) }
              else { t::lerp4(bg, t::lerp4(bg, [1.0; 4], 0.12), ht) };
    let tc = if enabled { t::lerp4(t::TEXT_SEC, t::TEXT, ht) } else { t::TEXT_DIM };
    dl.rounded_rect(x, y + 1.0, w, h, 6.0, [0.0, 0.0, 0.0, 0.12 * ht]);
    dl.rounded_rect(x, y, w, h, 6.0, col);
    if ht > 0.01 {
        dl.rounded_rect(x, y, w, h, 6.0, t::with_alpha(t::BORDER, ht * 0.5));
        dl.rounded_rect(x + 1.0, y + 1.0, w - 2.0, h - 2.0, 5.0, col);
    }
    fa.draw_centered(dl, label, x + w * 0.5, y + h * 0.5, t::FONT_NORMAL, tc);
    hov && inp.clicked()
}

pub fn button(
    dl: &mut DrawList, fa: &FontAtlas, inp: &InputState,
    x: f32, y: f32, w: f32, h: f32, label: &str, bg: [f32; 4], enabled: bool,
) -> bool {
    let hov = enabled && inp.in_rect(x, y, w, h);
    let pressed = hov && inp.mouse_down;
    let col = if pressed { t::lerp4(bg, [0.0;4], 0.2) }
              else if hov { t::lerp4(bg, [1.0;4], 0.1) } else { bg };
    let tc = if enabled { t::TEXT } else { t::TEXT_DIM };
    dl.rounded_rect(x, y, w, h, 6.0, col);
    fa.draw_centered(dl, label, x + w * 0.5, y + h * 0.5, t::FONT_NORMAL, tc);
    hov && inp.clicked()
}

#[derive(Default)]
pub struct ToggleState { anim_t: f32, hover_t: f32 }

pub fn toggle_anim(
    dl: &mut DrawList, inp: &InputState,
    state: &mut ToggleState, x: f32, y: f32, val: &mut bool,
) -> bool {
    let w = 40.0;
    let h = 22.0;
    let r = h * 0.5;
    let thumb_r = r - 3.0;

    // Smooth ON/OFF transition
    state.anim_t = input::smooth(
        state.anim_t,
        if *val { 1.0 } else { 0.0 },
        t::ANIM_NORMAL, inp.dt,
    );

    let hov = inp.in_rect(x - 2.0, y - 2.0, w + 4.0, h + 4.0);
    state.hover_t = input::smooth(
        state.hover_t,
        if hov { 1.0 } else { 0.0 },
        t::ANIM_FAST, inp.dt,
    );

    // Track colour: OFF = dark visible track, ON = accent
    let track_bg = t::lerp4(t::TOGGLE_OFF, t::ACCENT, state.anim_t);
    // Brighten slightly on hover
    let track_col = t::lerp4(track_bg, t::lerp4(track_bg, [1.0; 4], 0.08), state.hover_t);

    // Track shadow
    dl.rounded_rect(x, y + 1.0, w, h, r, [0.0, 0.0, 0.0, 0.15]);
    // Track fill
    dl.rounded_rect(x, y, w, h, r, track_col);
    // Track border — always visible, stronger when OFF
    let border_a = 0.14 - state.anim_t * 0.06;
    dl.rounded_rect(x, y, w, h, r, [1.0, 1.0, 1.0, border_a]);
    dl.rounded_rect(x + 1.0, y + 1.0, w - 2.0, h - 2.0, r - 1.0, track_col);

    // Thumb position
    let tx = x + r + (w - h) * state.anim_t;
    let ty = y + r;

    // Thumb shadow
    dl.circle(tx, ty + 1.0, thumb_r + 0.5, [0.0, 0.0, 0.0, 0.25]);
    // Thumb fill
    dl.circle(tx, ty, thumb_r, t::TEXT);
    // Thumb highlight (subtle specular dot at top)
    dl.circle(tx, ty - thumb_r * 0.35, thumb_r * 0.3, [1.0, 1.0, 1.0, 0.15]);

    let clicked = inp.clicked() && hov;
    if clicked { *val = !*val; }
    clicked
}

// ═══════════════════════════════════════════════════════════════
//  Keyboard Grid
// ═══════════════════════════════════════════════════════════════

pub struct HoveredKey { pub code: usize, pub name: &'static str, pub x: f32, pub y: f32, pub w: f32 }
pub struct GridResult { pub height: f32, pub hovered: Option<HoveredKey>, pub right_click: Option<(usize, f32, f32)> }

#[derive(Default)]
pub struct GridState {
    pub selected: HashSet<usize>,
    adding: bool,
    dragging: bool,
    hover_t: HashMap<usize, f32>,
    select_t: HashMap<usize, f32>,
}

impl GridState {
    pub fn select_all(&mut self, kl: &[Option<&str>]) {
        if self.selected.is_empty() {
            for (i, e) in kl.iter().enumerate() { if e.is_some() { self.selected.insert(i); } }
        } else { self.selected.clear(); }
    }
}

pub fn keyboard_grid(
    dl: &mut DrawList, fa: &FontAtlas, inp: &InputState,
    keys: &[Option<Key>], state: &mut GridState, show_act: bool,
    key_list: &[Option<&str>], x0: f32, y0: f32, avail_w: f32,
) -> GridResult {
    let cols = 16.0f32;
    let unit = (avail_w / cols).min(t::UNIT_PX);
    let row_h = unit * 0.84;
    let gap = t::KEY_GAP;
    let xoff = (avail_w - cols * unit) * 0.5;

    struct RK { code: usize, name: &'static str, rx: f32, ry: f32, rw: f32, rh: f32 }
    let mut rks = Vec::with_capacity(70);
    for (idx, entry) in key_list.iter().enumerate() {
        if let Some(name) = entry {
            if let Some(l) = get_key_layout(name) {
                rks.push(RK { code: idx, name: l.name,
                    rx: x0 + xoff + l.column * unit + gap * 0.5,
                    ry: y0 + l.row as f32 * (row_h + gap),
                    rw: l.width * unit - gap, rh: row_h });
            }
        }
    }

    // Input
    if inp.just_pressed() {
        for rk in &rks {
            if inp.in_rect(rk.rx, rk.ry, rk.rw, rk.rh) {
                state.adding = !state.selected.contains(&rk.code);
                state.dragging = true;
                break;
            }
        }
    }
    if !inp.mouse_down { state.dragging = false; }

    let mut result = GridResult { height: 5.0 * (row_h + gap), hovered: None, right_click: None };

    for rk in &rks {
        let hov = inp.in_rect(rk.rx, rk.ry, rk.rw, rk.rh);
        if state.dragging && hov {
            if state.adding { state.selected.insert(rk.code); }
            else { state.selected.remove(&rk.code); }
        }
        let is_sel = state.selected.contains(&rk.code);

        let ht = state.hover_t.entry(rk.code).or_insert(0.0);
        *ht = input::smooth(*ht, if hov { 1.0 } else { 0.0 }, t::ANIM_FAST, inp.dt);
        let hover_t = *ht;

        let st = state.select_t.entry(rk.code).or_insert(0.0);
        *st = input::smooth(*st, if is_sel { 1.0 } else { 0.0 }, t::ANIM_NORMAL, inp.dt);
        let select_t = *st;

        let pressed = inp.mouse_down && hov;
        let press_dy = if pressed { 1.0 } else { 0.0 };

        let base = match keys.get(rk.code) {
            Some(Some(k)) if k.color.r > 0 || k.color.g > 0 || k.color.b > 0 =>
                t::dim_key_color(k.color.r, k.color.g, k.color.b),
            _ => t::BG_KEY,
        };
        let bg = t::lerp4(t::lerp4(base, t::BG_KEY_HOVER, hover_t), t::BG_KEY_ACTIVE, select_t * 0.4);

        // Shadow (shrinks when pressed)
        let shadow_a = (0.20 + hover_t * 0.15) * (1.0 - press_dy * 0.5);
        dl.rounded_rect(rk.rx, rk.ry + 1.5 + press_dy, rk.rw, rk.rh, t::KEY_R, [0.0, 0.0, 0.0, shadow_a]);

        let ky = rk.ry + press_dy;

        // Selection glow
        if select_t > 0.01 {
            let e = 3.0 * select_t;
            dl.rounded_rect(rk.rx - e, ky - e, rk.rw + e * 2.0, rk.rh + e * 2.0,
                t::KEY_R + e, t::with_alpha(t::ACCENT, 0.15 * select_t));
        }

        // Fill
        dl.rounded_rect(rk.rx, ky, rk.rw, rk.rh, t::KEY_R, bg);

        // Top bevel
        dl.rect(rk.rx + 2.0, ky, rk.rw - 4.0, 1.0, [1.0, 1.0, 1.0, 0.04 + hover_t * 0.04]);

        // Selection border
        if select_t > 0.01 {
            let bw = 2.0;
            dl.rounded_rect(rk.rx, ky, rk.rw, rk.rh, t::KEY_R, t::with_alpha(t::ACCENT, 0.7 * select_t));
            dl.rounded_rect(rk.rx + bw, ky + bw, rk.rw - bw * 2.0, rk.rh - bw * 2.0,
                (t::KEY_R - bw).max(0.0), bg);
        }

        // Label
        let lb = (hover_t * 0.5 + select_t * 0.5).min(1.0);
        fa.draw_centered(dl, rk.name, rk.rx + rk.rw * 0.5, ky + rk.rh * 0.5, t::FONT_KEY,
            t::lerp4(t::TEXT_SEC, t::TEXT, lb));

        // Actuation overlays
        if show_act {
            if let Some(Some(k)) = keys.get(rk.code) {
                if k.down_actuation > 0.0 {
                    let s = format!("{:.2}", k.down_actuation);
                    let tw = fa.measure(&s, t::FONT_TINY).0;
                    fa.draw_text(dl, &s, rk.rx + (rk.rw - tw) * 0.5, ky + rk.rh - 12.0, t::FONT_TINY, t::PRESS_COL);
                }
                if k.up_actuation > 0.0 {
                    let s = format!("{:.2}", k.up_actuation);
                    let tw = fa.measure(&s, t::FONT_TINY).0;
                    fa.draw_text(dl, &s, rk.rx + (rk.rw - tw) * 0.5, ky + 2.0, t::FONT_TINY, t::RELEASE_COL);
                }
            }
        }

        // Tooltip info
        if hover_t > 0.5 {
            result.hovered = Some(HoveredKey { code: rk.code, name: rk.name,
                x: rk.rx + rk.rw * 0.5, y: rk.ry, w: rk.rw });
        }
        // Context menu
        if inp.right_clicked() && hov {
            result.right_click = Some((rk.code, inp.mouse_x, inp.mouse_y));
        }
    }

    result
}

// ═══════════════════════════════════════════════════════════════
//  HSV Colour Wheel — high-segment, smooth
// ═══════════════════════════════════════════════════════════════

#[derive(Default)]
pub struct HsvPickerState {
    pub hue: f32, pub sat: f32, pub val: f32,
    pub dragging_ring: bool, pub dragging_sv: bool,
}

impl HsvPickerState {
    pub fn is_dragging(&self) -> bool { self.dragging_ring || self.dragging_sv }
    pub fn sync_from_rgb(&mut self, r: u8, g: u8, b: u8) {
        let (h, s, v) = t::rgb_to_hsv(r as f32 / 255.0, g as f32 / 255.0, b as f32 / 255.0);
        self.hue = h; self.sat = s; self.val = v;
    }
    pub fn to_rgb(&self) -> (u8, u8, u8) {
        let [r, g, b] = t::hsv_to_rgb(self.hue, self.sat, self.val);
        ((r * 255.0).round() as u8, (g * 255.0).round() as u8, (b * 255.0).round() as u8)
    }
}

pub fn hsv_picker(
    dl: &mut DrawList, _fa: &FontAtlas, inp: &InputState,
    state: &mut HsvPickerState,
    cx: f32, cy: f32, radius: f32,
) -> bool {
    let r_out = radius;
    let ring_w = 12.0;
    let r_in = r_out - ring_w;

    // Number of segments — 180 = imperceptibly smooth hue ring
    let hue_seg = 180u32;
    // SV square subdivisions
    let sv_div = 28u32;

    // ── Dark backdrop behind the ring for depth ────────────────
    dl.circle(cx, cy, r_out + 1.0, [0.0, 0.0, 0.0, 0.20]);

    // ── SV square inside the ring ──────────────────────────────
    let half = r_in * 0.70;  // leave a tiny gap between square and ring
    let sq_x = cx - half;
    let sq_y = cy - half;
    let sq_size = 2.0 * half;
    let step = sq_size / sv_div as f32;

    for i in 0..sv_div {
        for j in 0..sv_div {
            let s0 = i as f32 / sv_div as f32;
            let s1 = (i + 1) as f32 / sv_div as f32;
            let v0 = 1.0 - j as f32 / sv_div as f32;
            let v1 = 1.0 - (j + 1) as f32 / sv_div as f32;
            let c00 = t::hsv_to_rgba(state.hue, s0, v0);
            let c10 = t::hsv_to_rgba(state.hue, s1, v0);
            let c01 = t::hsv_to_rgba(state.hue, s0, v1);
            let c11 = t::hsv_to_rgba(state.hue, s1, v1);
            dl.quad_colors(
                [sq_x + i as f32 * step,       sq_y + j as f32 * step],       c00,
                [sq_x + (i + 1) as f32 * step, sq_y + j as f32 * step],       c10,
                [sq_x + (i + 1) as f32 * step, sq_y + (j + 1) as f32 * step], c11,
                [sq_x + i as f32 * step,       sq_y + (j + 1) as f32 * step], c01,
            );
        }
    }

    // SV square subtle border
    let bdr = [1.0, 1.0, 1.0, 0.06];
    dl.rect(sq_x, sq_y, sq_size, 1.0, bdr);
    dl.rect(sq_x, sq_y + sq_size - 1.0, sq_size, 1.0, bdr);
    dl.rect(sq_x, sq_y, 1.0, sq_size, bdr);
    dl.rect(sq_x + sq_size - 1.0, sq_y, 1.0, sq_size, bdr);

    // ── Hue ring ───────────────────────────────────────────────
    for i in 0..hue_seg {
        let a0 = 2.0 * PI * i as f32 / hue_seg as f32;
        let a1 = 2.0 * PI * (i + 1) as f32 / hue_seg as f32;
        let h0 = 360.0 * i as f32 / hue_seg as f32;
        let h1 = 360.0 * (i + 1) as f32 / hue_seg as f32;
        let c0 = t::hsv_to_rgba(h0, 1.0, 1.0);
        let c1 = t::hsv_to_rgba(h1, 1.0, 1.0);
        let (sin0, cos0) = a0.sin_cos();
        let (sin1, cos1) = a1.sin_cos();
        dl.ring_segment(
            [cx + r_out * cos0, cy + r_out * sin0],
            [cx + r_out * cos1, cy + r_out * sin1],
            [cx + r_in * cos0,  cy + r_in * sin0],
            [cx + r_in * cos1,  cy + r_in * sin1],
            c0, c1,
        );
    }

    // Soft edge rings for cleaner outer/inner boundary
    dl.ring(cx, cy, r_out, r_out + 1.5, 96, [t::BG_RAISED[0], t::BG_RAISED[1], t::BG_RAISED[2], 0.7]);
    dl.ring(cx, cy, r_in - 1.5, r_in, 96, [t::BG_RAISED[0], t::BG_RAISED[1], t::BG_RAISED[2], 0.5]);

    // ── Hue indicator ──────────────────────────────────────────
    let ha = state.hue.to_radians();
    let r_mid = (r_out + r_in) * 0.5;
    let hx = cx + r_mid * ha.cos();
    let hy = cy + r_mid * ha.sin();
    dl.circle(hx, hy + 1.0, 6.0, [0.0, 0.0, 0.0, 0.35]);      // shadow
    dl.circle(hx, hy, 5.5, [1.0, 1.0, 1.0, 1.0]);              // outer ring
    dl.circle(hx, hy, 3.5, t::hsv_to_rgba(state.hue, 1.0, 1.0)); // hue colour fill

    // ── SV indicator ───────────────────────────────────────────
    let sx = sq_x + state.sat * sq_size;
    let sy = sq_y + (1.0 - state.val) * sq_size;
    dl.circle(sx, sy + 1.0, 6.0, [0.0, 0.0, 0.0, 0.35]);
    dl.circle(sx, sy, 5.5, [1.0, 1.0, 1.0, 1.0]);
    dl.circle(sx, sy, 3.5, t::hsv_to_rgba(state.hue, state.sat, state.val));

    // ── Interaction ────────────────────────────────────────────
    let dx = inp.mouse_x - cx;
    let dy = inp.mouse_y - cy;
    let dist = (dx * dx + dy * dy).sqrt();

    if inp.just_pressed() {
        if dist >= r_in - 2.0 && dist <= r_out + 4.0 {
            state.dragging_ring = true;
        } else if (inp.mouse_x - cx).abs() <= half + 3.0
               && (inp.mouse_y - cy).abs() <= half + 3.0
               && dist < r_in + 2.0 {
            state.dragging_sv = true;
        }
    }
    if !inp.mouse_down { state.dragging_ring = false; state.dragging_sv = false; }

    let mut changed = false;
    if state.dragging_ring {
        let a = dy.atan2(dx).to_degrees();
        state.hue = if a < 0.0 { a + 360.0 } else { a };
        changed = true;
    }
    if state.dragging_sv {
        state.sat = ((inp.mouse_x - sq_x) / sq_size).clamp(0.0, 1.0);
        state.val = 1.0 - ((inp.mouse_y - sq_y) / sq_size).clamp(0.0, 1.0);
        changed = true;
    }
    changed
}

// ═══════════════════════════════════════════════════════════════
//  Colour presets
// ═══════════════════════════════════════════════════════════════

pub fn color_presets(
    dl: &mut DrawList, fa: &FontAtlas, inp: &InputState,
    keys: &mut [Option<Key>], sel: &HashSet<usize>, x: f32, y: f32,
) -> bool {
    let presets: &[(&str, [u8; 3])] = &[
        ("R", [255,0,0]), ("G", [0,255,0]), ("B", [0,0,255]),
        ("W", [255,255,255]), ("Off", [0,0,0]),
    ];
    let mut changed = false;
    let mut cx = x;
    for (label, rgb) in presets {
        let bw = fa.measure(label, t::FONT_SMALL).0 + 14.0;
        let col = t::rgb_to_col(rgb[0].max(30), rgb[1].max(30), rgb[2].max(30));
        if button(dl, fa, inp, cx, y, bw, 22.0, label, col, true) {
            let nc = crate::model::key::KeyColor { r: rgb[0], g: rgb[1], b: rgb[2] };
            for &c in sel { if let Some(Some(k)) = keys.get_mut(c) { k.color = nc; } }
            changed = true;
        }
        cx += bw + 4.0;
    }
    changed
}

// ═══════════════════════════════════════════════════════════════
//  Tooltip

pub fn draw_tooltip(dl: &mut DrawList, fa: &FontAtlas, keys: &[Option<Key>], hk: &HoveredKey) {

unsafe extern "system" fn wnd_proc(hwnd: *mut c_void, msg: u32, wp: WPARAM, lp: LPARAM) -> LRESULT {

        WM_LBUTTONUP   => { INPUT.mouse_down = false; 0 }

        WM_RBUTTONUP   => { INPUT.right_down = false; 0 }
        WM_MOUSEWHEEL  => { INPUT.scroll_delta += ((wp >> 16) as i16) as f32 / 120.0; 0 }
        WM_SIZE => {
            let w = (lp & 0xFFFF) as u32;
            let h = ((lp >> 16) & 0xFFFF) as u32;
            if w > 0 && h > 0 { INPUT.width = w; INPUT.height = h; INPUT.resized = true; }
            0
        }
        WM_DESTROY => { PostQuitMessage(0); 0 }
        _ => DefWindowProcW(hwnd, msg, wp, lp)
    }
